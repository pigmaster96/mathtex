\chapter{Construction of the Real Numbers}
\begin{definition}
Let $S$ be a set. A \textbf{binary operation} on $S$ is a function $S\times S\to S$. A
\textbf{unary operation} on $S$ is a function $S\to S$.
\end{definition}
\noindent Let $S$ be a set and let $*:S\times S\to S$ be a binary operation. If $x,y\in S$, the correct way to write the result of the operation to the pair $\langle x,y\rangle$ would be
$*(\langle x,y\rangle)$. However since this is a bit cumbersome one may write $x*y$ instead.
Similarly if $\neg:S\to S$ is a unary operation and $x\in S$ one may choose to write $\neg x$ over $\neg(x)$.

\section{Axioms for the Natural numbers}
\subsection{Peano Postulates}
\textit{Axiom: Peano Postulates}\\
There exists a set $\mathbb N$ with an element $1\in\mathbb N$ and a function $s:\mathbb N\to\mathbb N$ that satisfy the following three properties.
\begin{itemize}
\item There is no $n\in\mathbb N$ such that $s(n)=1$.
\item The function $s$ is injective.
\item Let $G\subseteq\mathbb N$ be a set. Suppose that $1\in G$, and that if $g\in G$ then $s(g)\in G$. Then $G=\mathbb N$.
\end{itemize}
It will be shown that the set $\mathbb N$ is unique. We can therefore give it a name.
\begin{definition}
The set of \textbf{natural numbers} denoted $\mathbb N$, is the set the existence of which is given in the Peano Postulates.
\end{definition}

\subsection{Consequences of Peano Postulates}
\begin{lemma}\label{predecessor}
Let $a\in\mathbb N$. Suppose that $a\neq 1$. Then there is a unique $b\in\mathbb N$ such that $a=s(b)$. (where existence of $s$ and $\mathbb N$ are outlined by the peano axioms)
\end{lemma}
\begin{proof}
First we prove existence. Let 
\begin{equation*}
G=\{1\}\cup\{c\in\mathbb N\mid \exists b\in\mathbb N(s(b)=c)\}
\end{equation*}
(which exists by the subset, pairing, and union axioms) We will prove that $G=\mathbb N$. To do this we need to show $1\in G$, $G\subseteq\mathbb N$ and $\forall g\in G(s(g)\in G)$. 
The first two are 
straightforward. For the third suppose $g\in G$; we have two cases. First suppose $g\in\{1\}$; then $g=1\in\mathbb N$. Since $s:\mathbb N\to\mathbb N$, we then have $s(1)\in\mathbb N$.
So by definition $s(1)\in\{c\in\mathbb N\mid \exists b\in\mathbb N(s(b)=c)\}$ and $s(g)=s(1)\in G$.
Now suppose $g\in\{c\in\mathbb N\mid \exists b\in\mathbb N(s(b)=c)\}$; then by definition $g\in\mathbb N$ and $s(g)\in\mathbb N$ exists. 
By definition $s(g)\in\{c\in\mathbb N\mid \exists b\in\mathbb N(s(b)=c)\}$ so $s(g)\in G$. By the third peano postulate we then can assert that $G=\mathbb N$.
So $a\in\mathbb N=G$. Since $a\neq 1$ we must then have $a\in\{c\in\mathbb N\mid \exists b\in\mathbb N(s(b)=c)\}$ and by definition, $\exists b\in\mathbb N(s(b)=a)$.\\
\indent Now for uniqueness. Let $m$ and $n$ be arbitrary elements of $\mathbb N$ such that $a=s(m)$ and $a=s(n)$. Then $\langle m,a\rangle\in s$ and $\langle n,a\rangle\in s$. Since $s$ is injective
we must have $n=m$.
\end{proof}

\begin{theorem}\label{recursion}
Let $H$ be a set, Let $e\in H$ and let $k:H\to H$ be a function. Then there is a unique function $f:\mathbb N\to H$ such that $f(1)=e$ and $f\circ s=k\circ f$.
\end{theorem}
\begin{proof}
Letting $H, e$, and $k$ be arbitrary such that $e\in H$ and $k:H\to H$. We will show existence and uniqueness separately. We start with uniqueness.\\
\indent Let $g$ and $h$ be arbitrary such that $g:\mathbb N\to H$, $g(1)=e$, and $g\circ s=k\circ g$ and $h:\mathbb N\to H$, $h(1)=e$, and $h\circ s=k\circ h$. We will first define $V$ as the 
unique set (guaranteed by the subset and extensionality axioms) such that
\begin{equation*}
\forall a(a\in V\leftrightarrow a\in\mathbb N\land g(a)=h(a))
\end{equation*}
and show that $V=\mathbb N$. By peano's postulates, we need to show $V\subseteq\mathbb N$, $1\in V$, and $\forall n\in V(s(n)\in V)$. The first is straightforward. For the second requirement, we 
know by peano's postulates that $1\in\mathbb N$, so by our assumptions $g(1)$ and $h(1)$ exist and
$g(1)=e=h(1)$. So $1\in V$.\\
\indent For the third requirement, let $n$ be arbitrary and suppose that $n\in V$. Then by definition $n\in\mathbb N$ and $g(n)=h(n)$. Since $n\in\mathbb N=\text{dom}(h\circ s)$, we have, by our 
assumptions,
\begin{equation*}
h(s(n))=h\circ s(n)=k\circ h(n)=k(h(n))=k(g(n))=k\circ g(n)=g\circ s(n)=g(s(n))
\end{equation*}
we know by definition ran$(s)\subseteq\mathbb N$, so $s(n)\in\mathbb N$ and again by definition
$s(n)\in V$. We can conclude that $V=\mathbb N$.\\
\indent Suppose $\langle x,y\rangle\in g$, then $x\in\text{dom}g=\mathbb N=V$. So by definition $y=g(x)=h(x)$, and $\langle x,y\rangle\in h$.
Suppose $\langle x,y\rangle\in h$, then $x\in\text{dom}h=\mathbb N=V$. So by definition $y=h(x)=g(x)$, and $\langle x,y\rangle\in g$.\\
(next page)\newpage
Now for existence. We need to come up with a function from $\mathbb N$ into $H$ such that $f(1)=e$ and $f\circ s=k\circ f$. We first define $C$, where
\begin{equation*}
\forall X(X\in C\leftrightarrow X\subseteq\mathbb N\times H\land\langle 1,e\rangle\in X\land\forall n\forall y(\langle n,y\rangle\in X\to\langle s(n),k(y)\rangle\in X))
\end{equation*}
(guaranteed by the subset and extensionality axioms.) We propose that $f=\bigcap C$ is the desired function. To ensure its existence, we need to show that $C$ is nonempty.
We will show that $\mathbb N\times H\in C$. For the first requirement, clearly $\mathbb N\times H\subseteq\mathbb N\times H$.
Next, since $1\in\mathbb N$ and $e\in H$ then $\langle 1,e\rangle\in\mathbb N\times H$. Finally, let $n$ and $y$ be arbitrary such that $\langle n,y\rangle\in\mathbb N\times H$. 
Since $s:\mathbb N\to\mathbb N$ and $k:H\to H$, we must then have $s(n)\in\mathbb N$ and $k(y)\in H$. So $\langle s(n),k(y)\rangle\in\mathbb N\times H$. $f=\bigcap C$ is therefore a valid candidate.\\
\indent Before we prove the functionality of $f$, we first state a few facts about $f$.
First we note that $f\in C$. Let arbitrary $x\in\bigcap C$. Since $C$ is nonempty there exists $X\in C$, and by definition $x\in X\subseteq\mathbb N\times H$. So 
$\bigcap C\subseteq\mathbb N\times H$. Next let arbitrary $X\in C$. By definition $\langle 1,e\rangle\in X$, so $\langle 1,e\rangle\in\bigcap C$. Finally let $n$ and $y$ be arbitrary and suppose 
$\langle n,y\rangle\in\bigcap C$. We need to show that $\langle s(n),k(y)\rangle\in\bigcap C$. 
Let arbitrary $X\in C$. We have $\langle n,y\rangle\in X$. Since $X\in C$, by definition we must then have $\langle s(n),k(y)\rangle\in X$. We are done.\\
\indent Next we note that $\forall X\in C(f\subseteq X)$. This is straightforward to prove.
Finally we show
\begin{align*}
\forall n\forall y([\langle n,y\rangle&\in f\land\langle n,y\rangle\neq\langle 1,e\rangle]\\&\to
\exists m\exists u(\langle m,u\rangle\in f\land\langle n,y\rangle=\langle s(m),k(u)\rangle))
\end{align*}
Let $n$ and $y$ be arbitrary such that $\langle n,y\rangle\in f$ and $\langle n,y\rangle\neq\langle1,e\rangle$. Suppose, for contradiction, that the statement after the implication is false.
Then $\forall m\forall u(\langle m,u\rangle\in f\to \langle n,y\rangle\neq\langle s(m),k(u)\rangle)$. The set $f-\{\langle n,y\rangle\}$ exists by the subset axiom; we will show that this set is a
member of $C$. We know that $f\in C$, so $f\subseteq\mathbb N\times H$, and it is straightforward to show that $(f-\{\langle n,y\rangle\})\subseteq\mathbb N\times H$. 
Since $\langle 1,e\rangle\neq\langle n,y\rangle$ and $\langle 1,e\rangle\in f$, we also have $\langle 1,e\rangle\in f-\{\langle n,y\rangle\}$. 
Finally, let arbitrary $\langle a,b\rangle\in f-\{\langle n,y\rangle\}$. Then $\langle a,b\rangle\in f\in C$ and $\langle s(a),k(b)\rangle\in f$.
Since $\langle a,b\rangle\in f$, we must have, by our assumption, $\langle n,y\rangle\neq\langle s(a),k(b)\rangle$ and $\langle s(a),k(b)\rangle\in f-\{\langle n,y\rangle\}$. 
So $f-\{\langle n,y\rangle\}\in C$. It is straightforward to show that $f\nsubseteq f-\{\langle n,y\rangle\}$, contradicting the fact that $\forall X\in C(f\subseteq X)$.\\
\indent We now prove functionality. We will do this by defining $G$:
\begin{equation*}
\forall z(z\in G\leftrightarrow z\in\mathbb N\land\exists!x\in H(\langle z,x\rangle\in f))
\end{equation*}
(which exists by the subset axiom) and using the peano postulates to show that $G=\mathbb N$.\\
(next page)\newpage
We defined $G$:
\begin{equation*}
\forall z(z\in G\leftrightarrow z\in\mathbb N\land\exists!x\in H(\langle z,x\rangle\in f))
\end{equation*}
To show that $G=\mathbb N$, we need to show that $G\subseteq\mathbb N$, $1\in G$, and $\forall n\in G(s(n)\in G)$. The first requirement is straightforward. 
For the second, we already know by definition that $1\in\mathbb N$. Since $f\in C$ we know by definition that $\langle 1,e\rangle\in f$, where $e\in H$. 
Let arbitrary $x'\in H$ such that $\langle 1,x'\rangle\in f$. Suppose for contradiction that $e\neq x'$. Then we have $\langle 1,e\rangle\neq\langle 1,x'\rangle$; so by the third statement proven
earlier there exists $m,u$ such that $\langle m,u\rangle\in f$ and $\langle 1,x'\rangle=\langle s(m),k(u)\rangle$. 
This means that $1=s(m)$, where since $f\subseteq\mathbb N\times H$, $m\in\mathbb N$. This contradicts the peano postulates, which assert that 
$\neg\exists n\in\mathbb N(s(n)=1)$.\\
\indent For the third and final requirement, let $n$ be an arbitrary element of $G$. Then by definition $n\in\mathbb N$ and there exists a unique $x$ such that $\langle n,x\rangle\in f$. 
By definition we know $s(n)\in\mathbb N$, and 
since $f\in C$, we have $\langle s(n),k(x)\rangle\in f$, this verifies existence. Now let $x'$ be arbitrary such that $\langle s(n),x'\rangle\in f$. By peano's postulates we know that $s(n)\neq 1$,
so $\langle s(n),x'\rangle\neq\langle 1,e\rangle$. By the third statement proven earlier there exists $m,u$ such that $\langle m,u\rangle\in f$ and 
$\langle s(n),x'\rangle=\langle s(m),k(u)\rangle$. So $s(n)=s(m)$ and $x'=k(u)$. Since $s$ is injective we must have $m=n$. So $\langle n,u\rangle\in f$. By uniqueness of $x$ we then have
$u=x$. So $x'=k(u)=k(x)$, proving uniqueness. $s(n)\in G$.\\
\indent We have shown that $G=\mathbb N$. It follows from this that $f:\mathbb N\to H$. For functionality, let $x\in\text{dom}f$. Then there exists $y$ such that $\langle x,y\rangle\in f$.
Since $f\subseteq\mathbb N\times H$ we have $x\in\mathbb N=G$, and by definition $y$ is unique.
Next to show dom$f=\mathbb N$, first let arbitrary $x\in\text{dom}f$, similarly it immediately follows that $x\in\mathbb N$. Now let $x\in\mathbb N$. Then $x\in G$ and by definition there exists
$y\in H$ such that $\langle x,y\rangle\in f$, meaning $x\in\text{dom}f$. Finally to show ran$f\subseteq H$, let $y\in\text{ran}f$. 
Then there exists $x$ such that $\langle x,y\rangle\in f\subseteq\mathbb N\times H$, so $y\in H$.\\
\indent The fact that $\langle 1,e\rangle\in\bigcap C=f$ was already shown earlier. 
Finally we show $f\circ s=k\circ f$. Let $\langle x,y\rangle\in f\circ s$. Then $y=f\circ s(x)=f(s(x))$. Since $s(x)\neq 1$, $\langle s(x),y\rangle\neq\langle 1,e\rangle$ and as proven
earlier there then exists $m,u$ such that $\langle m,u\rangle\in f$ and $\langle s(x),y\rangle=\langle s(m),k(u)\rangle$. So $s(x)=s(m)$ and $y=k(u)$.
Since $s$ is injective $x=m$. We then have $\langle x,u\rangle\in f$ and $y=k(u)=k(f(x))=k\circ f(x)$. so $\langle x,y\rangle\in k\circ f$.
Next, suppose $\langle x,y\rangle\in k\circ f$.
By definition there exists $z$ such that $\langle x,z\rangle\in f$ and $\langle z,y\rangle\in k$.
Since $f\in C$ we have $\langle s(x),k(z)\rangle\in f$. We then have $y=k(z)=f(s(x)=f\circ s(x)$ and
$\langle x,y\rangle\in f\circ s$. We are done.
\end{proof}

\subsection{Addition on $\mathbb N$}
\begin{theorem}
There is a unique binary operation $+:\mathbb N\times\mathbb N\to\mathbb N$ that satisfies the following two properties for all $n,m\in\mathbb N$:
\begin{itemize}
\item $n+1=s(n)$
\item $n+s(m)=s(n+m)$
\end{itemize}
\end{theorem}
\begin{proof}
We start with uniqueness. Let $+$ and $\oplus$ be arbitrary such that $+,\oplus:\mathbb N\times\mathbb N\to\mathbb N$, 
$\forall m\in\mathbb N\forall n\in\mathbb N(+(\langle n,1\rangle)=s(n)\land+(\langle n,s(m)\rangle)=s(+(\langle n,m\rangle)))$, and 
$\forall m\in\mathbb N\forall n\in\mathbb N(\oplus(\langle n,1\rangle)=s(n)\land\oplus(\langle n,s(m)\rangle)=s(\oplus(\langle n,m\rangle)))$. We need to show $+=\oplus$. 
We define $G$ as
\begin{equation*}
\forall x(x\in G\leftrightarrow x\in\mathbb N\land\forall n\in\mathbb N(+(\langle n,x\rangle)=\oplus(\langle n,x\rangle)))
\end{equation*}
(guaranteed by the subset axiom.) We will show that $G=\mathbb N$. To do this we need to show
$G\subseteq\mathbb N$, $1\in G$, and $\forall g\in G(s(g)\in G)$. The first requirement is straightforward. For the second requirement, we already know $1\in\mathbb N$, and for arbitrary 
$n\in\mathbb N$, $+(\langle n,1\rangle)=s(n)=\oplus(\langle n,1\rangle)$.\\
\indent For the third requirement, let arbitrary $g\in G$. Then by definition $g\in\mathbb N$ and $\forall n\in\mathbb N(+(\langle n,g\rangle)=\oplus(\langle n,g\rangle))$.
Since $s:\mathbb N\to\mathbb N$ we know that $s(g)\in\mathbb N$ exists. Let $n$ be arbitrary. By definition
$+(\langle n,s(g)\rangle)=s(+(\langle n,g\rangle))$ and $\oplus(\langle n,s(g)\rangle)=s(\oplus(\langle n,g\rangle))$. 
By assumption we can assert that $+(\langle n,g\rangle)=\oplus(\langle n,g\rangle)$, so 
$+(\langle n,s(g)\rangle)=s(+(\langle n,g\rangle))=s(\oplus(\langle n,g\rangle))=\oplus(\langle n,s(g)\rangle)$. We conclude that $s(g)\in G$ and $G=\mathbb N$.\\
\indent Let $\langle\langle x,y\rangle,z\rangle\in+$. By definition $y\in\mathbb N=G$ and $x\in\mathbb N$. So $+(\langle x,y\rangle)=\oplus(\langle x,y\rangle)$ and $z=\oplus(\langle x,y\rangle)$.
So $\langle\langle x,y\rangle,z\rangle\in\oplus$. Now let $\langle\langle x,y\rangle,z\rangle\in\oplus$. A similar argument shows that $\langle\langle x,y\rangle,z\rangle\in+$.\\
\indent Now for existence. We want to show
\begin{align*}
\exists+(+:\mathbb N\times\mathbb N\to\mathbb N\land\forall m\in\mathbb N\forall&n\in\mathbb N
(+(\langle n,1\rangle)=s(n)\\
&\land+(\langle n,s(m)\rangle)=s(+(\langle n,m\rangle))))
\end{align*}
Theorem \ref{recursion} allows us to assert
\begin{equation*}
\forall k\forall H\forall e([e\in H\land k:H\to H]\to\exists!f(f:\mathbb N\to H\land f(1)=e\land f\circ s=k\circ f))
\end{equation*}
It immediately follows that
\begin{equation*}
\forall p\in\mathbb N\exists!f(f:\mathbb N\to\mathbb N\land f(1)=p\land f\circ s=s\circ f)
\end{equation*}
We will define $+$ as the set where
\begin{align*}
\forall z(z\in+\leftrightarrow\exists a\in\mathbb N\exists b\in\mathbb N(\exists f(&f:\mathbb N\to\mathbb N\land f(1)=s(a)\land f\circ s=s\circ f\\&\land\exists c(\langle b,c\rangle\in f\land 
z=\langle\langle a,b\rangle,c\rangle))))
\end{align*}
Existence follows from the subset axiom, where it can be shown that $+\subseteq(\mathbb N\times\mathbb N)\times\mathbb N$.\\
(next page)\newpage
We defined $+$ as
\begin{align*}
\forall z(z\in+\leftrightarrow\exists a\in\mathbb N\exists b\in\mathbb N(\exists f(&f:\mathbb N\to\mathbb N\land f(1)=s(a)\land f\circ s=s\circ f\\&\land\exists c(\langle b,c\rangle\in f\land 
z=\langle\langle a,b\rangle,c\rangle))))
\end{align*}
First we prove functionality. Let $x$ be an arbitrary element of dom$(+)$. Then there exists $y$ such that $\langle x,y\rangle\in +$. Let $y'$ be arbitrary such that $\langle x,y'\rangle\in+$.
$\langle x,y\rangle\in +$ by definition means that there exists $a,b\in\mathbb N$, and $f:\mathbb N\to\mathbb N$ such that $f(1)=s(a)$ and $f\circ s=s\circ f$, and that there exists $c$ such that
$\langle b,c\rangle\in f$ and $\langle x,y\rangle=\langle\langle a,b\rangle,c\rangle$.
$\langle x,y'\rangle\in +$ similarly implies the existence of $a_0,b_0\in\mathbb N$ and $f_0:\mathbb N\to\mathbb N$ such that $f_0(1)=s(a_0)$ and $f_0\circ s=s\circ f_0$, and the existence of $c_0$ 
such that $\langle b_0,c_0\rangle\in f_0$ and $\langle x,y'\rangle=\langle\langle a_0,b_0\rangle,c_0\rangle$. Then $\langle a,b\rangle=x=\langle a_0,b_0\rangle$, so $a_0=a$ and $b_0=b$.
It follows that $s(a_0)=s(a)\in\mathbb N$. This and 
\begin{equation*}
\forall p\in\mathbb N\exists!f(f:\mathbb N\to\mathbb N\land f(1)=p\land f\circ s=s\circ f)
\end{equation*}
implies that $f=f_0$. So we have $y=c=f(b)=f(b_0)=f_0(b_0)=c_0=y'$.\\
\indent Next we show that dom$(+)=\mathbb N\times\mathbb N$. First let $x$ be an arbitrary element of dom$(+)$. Then there exists $y$ such that $\langle x,y\rangle\in+$. By definition 
there then exists $a,b\in\mathbb N$ and there exists $c$ such that $\langle x,y\rangle=\langle\langle a,b\rangle,c\rangle$. So $x=\langle a,b\rangle\in\mathbb N\times\mathbb N$.\\
\indent Now suppose $x\in\mathbb N\times\mathbb N$. Then by definition there exists $a,b\in\mathbb N$ such that $x=\langle a,b\rangle$. $s(a)\in\mathbb N$ exists, so by theorem \ref{recursion} there
exists $f$ such that $f:\mathbb N\to\mathbb N$, $f(1)=s(a)$, and $f\circ s=s\circ f$.
Since $b\in\mathbb N=\text{dom}f$, $f(b)$ exists and $\langle b,f(b)\rangle\in f$.
Then by definition, $\langle\langle a,b\rangle,f(b)\rangle\in+$, so $x=\langle a,b\rangle\in\text{dom}(+)$.\\
\indent Next we show that ran$(+)\subseteq\mathbb N$. Let arbitrary $y\in\text{ran}(+)$. Then there exists $x$ such that $\langle x,y\rangle\in+$.
By definition there then exists $a,b\in\mathbb N$, $f:\mathbb N\to\mathbb N$, and $c$ such that
$\langle b,c\rangle\in f$ and $\langle x,y\rangle=\langle\langle a,b\rangle,c\rangle$.
So we have $y=c=f(b)\in\mathbb N$.\\
\indent Now we verify the two properties of addition, that is,
\begin{equation*}
\forall m\in\mathbb N\forall n\in\mathbb N(+(\langle n,1\rangle)=s(n)\land+(\langle n,s(m)\rangle)=s(+(\langle n,m\rangle)))
\end{equation*}
Let arbitrary $n,m\in\mathbb N$. We know that $+:\mathbb N\times\mathbb N\to\mathbb N$, so $\langle n,1\rangle\in\mathbb N\times\mathbb N=\text{dom}(+)$ and there exists $y$ such that
$\langle\langle n,1\rangle,y\rangle\in+$. By definition there then exists $a,b\in\mathbb N$ and $f:\mathbb N\to\mathbb N$ such that $f(1)=s(a)$, and there exists $c$ such that 
$\langle b,c\rangle\in f$ and $\langle\langle n,1\rangle,y\rangle=\langle\langle a,b\rangle,c\rangle$. 
Then $n=a$, $1=b$, and $y=c$. We then have $\langle 1,y\rangle\in f$ and $y=f(1)=s(a)=s(n)$.
We conclude that $\langle\langle n,1\rangle,s(n)\rangle\in+$ and $+(\langle n,1\rangle)=s(n)$.\\
(next page)\newpage
Now for the second property. We know that $s(m)\in\mathbb N$. So $\langle n,s(m)\rangle\in\mathbb N\times\mathbb N=\text{dom}(+)$ and there exists $y$ such that 
$\langle\langle n,s(m)\rangle,y\rangle\in+$. By definition there then exists $a,b\in\mathbb N$ such that $f:\mathbb N\to\mathbb N$, $f(1)=s(a)$, and $f\circ s=s\circ f$, and there 
exists $c$ such that $\langle b,c\rangle\in f$ and $\langle\langle n,s(m)\rangle,y\rangle=\langle\langle a,b\rangle,c\rangle$. 
We then have $n=a$, $s(m)=b$, and $y=c=f(b)=f(s(m))=s(f(m))$. We know that $\langle n,m\rangle\in\mathbb N\times\mathbb N=\text{dom}(+)$, so there exists $y_0$ such that 
$\langle\langle n,m\rangle,y_0\rangle\in+$. Again by definition there exists $a_0,b_0\in\mathbb N$ and $f_0:\mathbb N\to\mathbb N$ such that $f_0(1)=s(a_0)$ and $f_0\circ s=s\circ f_0$ and there 
exists $c_0$ such that $\langle b_0,c_0\rangle\in f_0$
and $\langle\langle n,m\rangle,y_0\rangle=\langle\langle a_0,b_0\rangle,c_0\rangle$. Since it follows that $a_0=n=a$ and $s(a_0)=s(a)$, theorem \ref{recursion} allows us to assert $f=f_0$.
Putting it all together, we have $+(\langle n,m\rangle)=y_0=c_0=f_0(b_0)=f_0(m)=f(m)$.
We conclude that $+(\langle n,s(m)\rangle)=y=s(f(m))=s(+(\langle n,m\rangle))$.
\end{proof}

\subsection{Multiplication on $\mathbb N$}
\begin{theorem}
There is a unique binary operation $\cdot:\mathbb N\times\mathbb N\to\mathbb N$ that satisfies the following two properties for all $n,m\in\mathbb N$:
\begin{itemize}
\item $n\cdot1=n$
\item $n\cdot s(m)=(n\cdot m)+n$
\end{itemize}
\end{theorem}
\begin{proof} We start with uniqueness.
Let $\cdot,\odot$ be arbitrary. Suppose $\cdot:\mathbb N\times\mathbb N\to\mathbb N$ and 
$\forall m\in\mathbb N\forall n\in\mathbb N(\cdot(\langle n,1\rangle)=n\land\cdot(\langle n,s(m)\rangle)=+(\langle\cdot(\langle n,m\rangle),n\rangle))$. Similarly suppose 
$\odot:\mathbb N\times\mathbb N\to\mathbb N$ and 
$\forall m\in\mathbb N\forall n\in\mathbb N(\odot(\langle n,1\rangle)=n\land\odot(\langle n,s(m)\rangle)=+(\langle\odot(\langle n,m\rangle),n\rangle))$.
We want to show that $\cdot=\odot$. We do this by defining $G$ such that
\begin{equation*}
\forall z(z\in G\leftrightarrow z\in\mathbb N\land\forall n\in\mathbb N(\cdot(\langle n,z\rangle)=\odot(\langle n,z\rangle)))
\end{equation*}
(where existence is assured by the subset axiom) and showing that $G=\mathbb N$.\\
\indent We need to show $G\subseteq\mathbb N$, $1\in G$, and $\forall g\in G(s(g)\in G)$.
The first requirement is straightforward. For the second, we already know $1\in\mathbb N$, and see that for arbitrary $n\in\mathbb N$ we have 
$\cdot(\langle n,1\rangle)=n=\odot(\langle n,1\rangle)$.\\
\indent For the third requirement let arbitrary $g\in G$. Then by definition $g\in\mathbb N$ and $\forall n\in\mathbb N(\cdot(\langle n,g\rangle)=\odot(\langle n,g\rangle))$.
We know that $s(g)\in\mathbb N$ exists. So since $\langle n,s(g)\rangle\in\mathbb N\times\mathbb N=\text{dom}(\cdot)$, we have, by our assumptions, 
$\cdot(\langle n,s(g)\rangle)=+(\langle\cdot(\langle n,g\rangle),n\rangle)=+(\langle\odot(\langle n,g\rangle),n\rangle)=\odot(\langle n,s(g)\rangle)$. So $s(g)\in G$ and $G=\mathbb N$.\\
\indent To prove the desired result, let arbitrary $z\in\cdot\subseteq(\mathbb N\times\mathbb N)\times\mathbb N$. Then there exists $a_0,a_1,b\in\mathbb N$ such that 
$\langle\langle a_0,a_1\rangle,b\rangle=z\in\cdot$. So $b=\cdot(\langle a_0,a_1\rangle)$.
Since $a_1\in\mathbb N=G$, we can then conclude that $b=\cdot(\langle a_0,a_1\rangle)=\odot(\langle a_0,a_1\rangle)$. So $z=\langle\langle a_0,a_1\rangle,b\rangle\in\odot$. Next suppose $z\in\odot$.
It then follows from a similar argument that $z\in\cdot$. We are done.\\
(next page)\newpage
Now for existence. We start by defining a function $h$ for all $q$, where
\begin{equation*}
\forall q\exists h\forall z(z\in h\leftrightarrow\exists m\in\mathbb N\exists x\in\mathbb N(z=\langle m,x\rangle\land\langle\langle m,q\rangle,x\rangle\in+))
\end{equation*}
(which exists by the subset axiom, where $\mathbb N\times\mathbb N$ is a bounding set. Uniqueness for each $q$ follows from extensionality.) We show that for all $q\in\mathbb N$, this $h$ is a 
function from $\mathbb N$ into $\mathbb N$. That is
\begin{align*}
\forall q\in\mathbb N[\forall z(z\in h\leftrightarrow\exists m\in\mathbb N\exists x\in\mathbb N(z=\langle m,x\rangle\land\langle\langle m,q\rangle,x\rangle\in+))\to h:\mathbb N\to\mathbb N]
\end{align*}
Let arbitrary $q\in\mathbb N$. Suppose that the statement before the implication is true.
First to prove functionality. Let $a\in\text{dom}h$. Then there exists $b$ such that $\langle a,b\rangle\in h$. Let $b'$ be arbitrary such that $\langle a,b'\rangle\in h$. By our assumptions we 
must have $a,b,b'\in\mathbb N$, $+(\langle a,q\rangle)=b$, and $+(\langle a,q\rangle)=b'$. Since
$+:\mathbb N\times\mathbb N\to\mathbb N$ we must have $b=b'$. Next for the domain bound suppose arbitrary $a\in\text{dom}h$. Then there exists $b$ such that $\langle a,b\rangle\in h$. It immediately
follows that $a\in\mathbb N$. Suppose instead that $a\in\mathbb N$. Then $\langle a,q\rangle\in\mathbb N\times\mathbb N=\text{dom}(+)$.
So $+(\langle a,q\rangle)\in\mathbb N$ exists and by definition, $\langle a,+(\langle a,q\rangle)\rangle\in h$. So $a\in\text{dom}h$. Finally for the range bound, let arbitrary $b\in\text{ran}h$.
Then there exists $a$ such that $\langle a,b\rangle\in h$. It immediately follows by definition that $b\in\mathbb N$. We have therefore shown
\begin{align*}
\forall q\in\mathbb N\exists!h
(\forall z(z\in h\leftrightarrow\exists&m\in\mathbb N\exists x\in\mathbb N(z=\langle m,x\rangle\land\langle\langle m,q\rangle,x\rangle\in+))\\&\land h:\mathbb N\to\mathbb N)
\end{align*}
Theorem \ref{recursion} states
\begin{equation*}
\forall H\forall q\forall h([q\in H\land h:H\to H]\to\exists!g(g:\mathbb N\to H\land g(1)=q\land g\circ s=h\circ g))
\end{equation*}
which allows us to assert
\begin{equation*}
\forall q\forall h([q\in\mathbb N\land h:\mathbb N\to\mathbb N]\to\exists!g(g:\mathbb N\to\mathbb N\land g(1)=q\land g\circ s=h\circ g))
\end{equation*}

We showed
\begin{align*}
\forall q\in\mathbb N\exists!h
(\forall z(z\in h\leftrightarrow\exists&m\in\mathbb N\exists x\in\mathbb N(z=\langle m,x\rangle\land\langle\langle m,q\rangle,x\rangle\in+))\\&\land h:\mathbb N\to\mathbb N)
\end{align*}
and
\begin{equation*}
\forall q\forall h([q\in\mathbb N\land h:\mathbb N\to\mathbb N]\to\exists!g(g:\mathbb N\to\mathbb N\land g(1)=q\land g\circ s=h\circ g))
\end{equation*}
Now we construct $\cdot$ as the set
\begin{align*}
\forall z(z\in\cdot\leftrightarrow\exists&a\in\mathbb N\exists b\in\mathbb N(\\
\exists h(&\forall z_0(z_0\in h\leftrightarrow\exists m\in\mathbb N\exists x\in\mathbb N(z_0=\langle m,x\rangle\land\langle\langle m,a\rangle,x\rangle\in+))\\
&\land h:\mathbb N\to\mathbb N\land\exists g(g:\mathbb N\to\mathbb N\land g(1)=a\land g\circ s=h\circ g\\
&\land\exists c(\langle b,c\rangle\in g\land z=\langle\langle a,b\rangle,c\rangle)))))
\end{align*}
(existence follows from the subset axiom, where $(\mathbb N\times\mathbb N)\times\mathbb N$ is a bounding set.) We want to verify
\begin{equation*}
\cdot:\mathbb N\times\mathbb N\to\mathbb N\land\forall m\in\mathbb N\forall n\in\mathbb N(\cdot(\langle n,1\rangle)=n\land\cdot(\langle n,s(m)\rangle)=+(\langle\cdot(\langle n,m\rangle),n\rangle))
\end{equation*}
We start with functionality. Let arbitrary $p\in\text{dom}(\cdot)$. Then by definition there exists $q$ such that $\langle p,q\rangle\in\cdot$. This means
that there exists $a,b\in\mathbb N$ and $h:\mathbb N\to\mathbb N$ such that 
$\forall z_0(z_0\in h\leftrightarrow\exists m\in\mathbb N\exists x\in\mathbb N(z_0=\langle m,x\rangle\land\langle\langle m,a\rangle,x\rangle\in+))$. 
There also exists $g:\mathbb N\to\mathbb N$ such that $g(1)=a$ and $g\circ s=h\circ g$, and $c$ such that $\langle b,c\rangle\in g$ and $\langle p,q\rangle=\langle\langle a,b\rangle,c\rangle$.
Now suppose there exists $\langle p,q'\rangle\in\cdot$. 
Similarly there exists $a',b'\in\mathbb N$, $h':\mathbb N\to\mathbb N$ such that 
$\forall z_0(z_0\in h'\leftrightarrow\exists m\in\mathbb N\exists x\in\mathbb N(z_0=\langle m,x\rangle\land\langle\langle m,a'\rangle,x\rangle\in+))$, 
$g':\mathbb N\to\mathbb N$ such that $g'(1)=a'$ and $g'\circ s=h'\circ g'$, and $c'$ such that $\langle b',c'\rangle\in g'$ and $\langle p,q'\rangle=\langle\langle a',b'\rangle,c'\rangle$. 
We have $\langle a',b'\rangle=p=\langle a,b\rangle$. So $a'=a$ and $b'=b$. Uniqueness, as proven, implies that $h=h'$ and $g=g'$. So
$q'=c'=g'(b')=g(b')=g(b)=c=q$.\\
\indent Next we show dom$(\cdot)=\mathbb N\times\mathbb N$. Let arbitrary $p\in\text{dom}(\cdot)$.
Then by definition there exists $q$ such that $\langle p,q\rangle\in\cdot$.
So there exists $a,b\in\mathbb N$ and $c$ such that $\langle p,q\rangle=\langle\langle a,b\rangle,c\rangle$.
So $p=\langle a,b\rangle\in\mathbb N\times\mathbb N$. Now let $p\in\mathbb N\times\mathbb N$. Then by definition there exists $a,b\in\mathbb N$ such that $p=\langle a,b\rangle$. Since 
$a\in\mathbb N$, as proven there exists $h:\mathbb N\to\mathbb N$ such that 
$\forall z(z\in h\leftrightarrow\exists m\in\mathbb N\exists x\in\mathbb N(z=\langle m,x\rangle\land\langle\langle m,a\rangle,x\rangle\in+))$, 
and there exists $g:\mathbb N\to\mathbb N$ such that $g(1)=a$ and $g\circ s=h\circ g$.
By definition we then have $\langle\langle a,b\rangle,g(b)\rangle\in\cdot$ and $p=\langle a,b\rangle\in\text{dom}(\cdot)$.\\
\indent For the range bound, let $q\in\text{ran}(\cdot)$. Then there exists $p$ such that $\langle p,q\rangle\in\cdot$. So there exists $a,b\in\mathbb N$, $g:\mathbb N\to\mathbb N$, and $c$ such 
that $\langle b,c\rangle\in g$ and $\langle p,q\rangle=\langle\langle a,b\rangle,c\rangle$.
So $q=c=g(b)\in\mathbb N$.\\
\indent Finally we verify the two properties of multiplication. Let arbitrary $m,n\in\mathbb N$.
We first show $\cdot(\langle n,1\rangle)=n$. We know that $\langle\langle n,1\rangle,\cdot(\langle n,1\rangle)\rangle\in\cdot$.
By definition there then exists $a,b\in\mathbb N$, $g:\mathbb N\to\mathbb N$ such that $g(1)=a$, and $c$ such that $\langle b,c\rangle\in g$ and 
$\langle\langle n,1\rangle,\cdot(\langle n,1\rangle)\rangle=\langle\langle a,b\rangle,c\rangle$.
So $\cdot(\langle n,1\rangle)=c=g(b)=g(1)=a=n$.

Now we show $\cdot(\langle n,s(m)\rangle)=+(\langle\cdot(\langle n,m\rangle),n\rangle)$.
$s(m)\in\mathbb N$ exists, so we have $\langle\langle n,s(m)\rangle,\cdot(\langle n,s(m)\rangle)\rangle\in\cdot$. Then there exists $a,b\in\mathbb N$, $h:\mathbb N\to\mathbb N$ such that 
$\forall z_0(z_0\in h\leftrightarrow\exists m_0\in\mathbb N\exists x\in\mathbb N(z_0=\langle m_0,x\rangle\land\langle\langle m_0,a\rangle,x\rangle\in+))$, $g:\mathbb N\to\mathbb N$ such that 
$g(1)=a$ and $g\circ s=h\circ g$, and $c$ such that $\langle b,c\rangle\in g$ and 
$\langle\langle n,s(m)\rangle,\cdot(\langle n,s(m)\rangle)\rangle=\langle\langle a,b\rangle,c\rangle$.
We then have $\cdot(\langle n,s(m)\rangle)=c=g(b)=g(s(m))=h(g(m))$ and $a=n$. Similarly we know that $\langle\langle n,m\rangle,\cdot(\langle n,m\rangle)\rangle\in\cdot$.
So there exists $a_0,b_0\in\mathbb N$, $h_0:\mathbb N\to\mathbb N$ such that 
$\forall z_0(z_0\in h_0\leftrightarrow\exists m_0\in\mathbb N\exists x\in\mathbb N(z_0=\langle m_0,x\rangle\land\langle\langle m_0,a_0\rangle,x\rangle\in+))$,
$g_0:\mathbb N\to\mathbb N$ such that $g_0(1)=a_0$ and $g_0\circ s=h_0\circ g_0$, and $c_0$ such 
that $\langle b_0,c_0\rangle\in g_0$ and $\langle\langle n,m\rangle,\cdot(\langle n,m\rangle)\rangle=\langle\langle a_0,b_0\rangle,c_0\rangle$.
This means that $a_0=n=a$, and that by our earlier statements we can assert that $h=h_0$, and $g=g_0$. So $\cdot(\langle n,m\rangle)=c_0=g_0(b_0)=g(m)$. We then have
$\cdot(\langle n,s(m)\rangle)=h(g(m))=h(\cdot(\langle n,m\rangle))$. From the definition of $h$, there then exists $m_0,x\in\mathbb N$ such that
$\langle\cdot(\langle n,m\rangle),h(\cdot(\langle n,m\rangle))\rangle=\langle m_0,x\rangle$ and
$\langle\langle m_0,a\rangle,x\rangle\in+$. This and the fact that $a=n$ allows us to conclude
$+(\langle\cdot(\langle n,m\rangle),n\rangle)=h(\cdot(\langle n,m\rangle))=\cdot(\langle n,s(m)\rangle)$.
\end{proof}

\subsection{Properties of addition and multiplication on the naturals}
\begin{theorem}\label{natural arithmetic}
Let $a,b,c\in\mathbb N$.
\begin{enumerate}
\item If $a+c=b+c$, then $a=b$\quad(Cancellation Law for Addition)
\item $(a+b)+c=a+(b+c)$\quad(Associative Law for Addition)
\item $1+a=s(a)=a+1$
\item $a+b=b+a$\quad(Commutative Law for Addition)
\item $a+b\neq1$
\item $a+b\neq a$
\item $a\cdot1=a=1\cdot a$\quad(Identity Law for Multiplication)
\item $(a+b)c=ac+bc$\quad(Distributive Law)
\item $ab=ba$\quad(Commutative Law for Multiplication)
\item $c(a+b)=ca+cb$\quad(Distributive Law)
\item $(ab)c=a(bc)$\quad(Associative Law for Multiplication)
\item If $ac=bc$ then $a=b$\quad(Cancellation Law for Multiplication)
\item $ab=1$ iff $a=1=b$
\end{enumerate}
\end{theorem}
\noindent(next page)\newpage
\begin{proof}(1)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N(a+c=b+c\to a=b)
\end{equation*}
To do this we define the set $G$ as the set where
\begin{equation*}
\forall c(c\in G\leftrightarrow c\in\mathbb N\land\forall a\in\mathbb N\forall b\in\mathbb N(a+c=b+c\to a=b))
\end{equation*}
(where existence follows from the subset axiom using $\mathbb N$ as the bounding set. Uniqueness follows from extensionality.) Then we show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward; we need to show $1\in G$ and $\forall g\in G(s(g)\in G)$.
For the former, we already know $1\in\mathbb N$. Let arbitrary $a,b\in\mathbb N$. Suppose that $a+1=b+1$. By definition we then have $s(a)=s(b)$, so by injectivity $a=b$ and $1\in G$.\\
\indent Now the latter. Let arbitrary $g\in G$. As defined we then have $g\in\mathbb N$ and $\forall a\in\mathbb N\forall b\in\mathbb N(a+g=b+g\to a=b)$. $s(g)\in\mathbb N$ exists. 
Let arbitrary $a,b\in\mathbb N$ and suppose that $a+s(g)=b+s(g)$. Then by the definition of addition we have $s(a+g)=s(b+g)$ and, by injectivity, $a+g=b+g$. Since $g\in G$ we then have
$a=b$. So $G=\mathbb N$. To prove the initial statement, let arbitrary $c\in\mathbb N$. Then from $\mathbb N=G$ the desired conclusion immediately follows.
\end{proof}
\begin{proof}(2) We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N((a+b)+c=a+(b+c)) 
\end{equation*}
To do this we define $G$ as the set where
\begin{equation*}
\forall c(c\in G\leftrightarrow c\in\mathbb N\land\forall a\in\mathbb N\forall b\in\mathbb N((a+b)+c=a+(b+c)))
\end{equation*}
(where existence and uniqueness can be easily shown) and show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward. We need to show $1\in G$ and $\forall g\in G(s(g)\in G)$.\\
\indent For the former, we already know $1\in\mathbb N$. Let arbitrary $a,b\in\mathbb N$. By the definition of addition we have
\begin{equation*}
(a+b)+1=s(a+b)=a+s(b)=a+(b+1)
\end{equation*}
\indent Now the latter. Let arbitrary $g\in G$. Then by definition $g\in\mathbb N$ and $\forall a\in\mathbb N\forall b\in\mathbb N((a+b)+g=a+(b+g))$. We know $s(g)\in\mathbb N$ exists. 
Let arbitrary $a,b\in\mathbb N$. By definition we then have
\begin{equation*}
(a+b)+s(g)=s((a+b)+g)=s(a+(b+g))=a+s(b+g)=a+(b+s(g))
\end{equation*}
So $\mathbb N=G$. To prove the initial statement, let arbitrary $c\in\mathbb N=G$. The desired conclusion immediately follows.
\end{proof}

\begin{proof}(3)
We want to show
\begin{equation*}
\forall a\in\mathbb N(1+a=s(a)=a+1)
\end{equation*}
To do this we define $G$ as the set where
\begin{equation*}
\forall a(a\in G\leftrightarrow a\in\mathbb N\land 1+a=s(a)=a+1)
\end{equation*}
(Showing existence and uniqueness is straightforward.) We show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward. We need to show that $1\in G$ and $\forall g\in G(s(g)\in G)$. For the former we already know that $1\in\mathbb N$, and by definition we have
$1+1=s(1)$.\\
\indent Now the latter. Let arbitrary $g\in G$. Then by definition $g\in\mathbb N$ and $1+g=s(g)=g+1$. We know $s(g)\in\mathbb N$ exists. We then have
\begin{equation*}
1+s(g)=s(1+g)=s(s(g))=s(g+1)=(g+1)+1=s(g)+1
\end{equation*}
So $G=\mathbb N$. Proving the initial statement is straightforward. Let arbitrary $a\in\mathbb N=G$ and the desired conclusion immediately follows.
\end{proof}
\begin{proof}(4)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a+b=b+a)
\end{equation*}
To do this we define $G$ as the set such that
\begin{equation*}
\forall a(a\in G\leftrightarrow a\in\mathbb N\land\forall b\in\mathbb N(a+b=b+a))
\end{equation*}
(Showing existence and uniqueness is straightforward.) We show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward. We need to show that $1\in G$ and $\forall g\in G(s(g)\in G)$. For the former we already know $1\in\mathbb N$. 
Letting arbitrary $b\in\mathbb N$, by (3) we have $1+b=b+1$.\\
\indent Now the latter. Let arbitrary $g\in G$. Then by definition $g\in\mathbb N$ and $\forall b\in\mathbb N(g+b=b+g)$. $s(g)\in\mathbb N$ exists. Let arbitrary $b\in\mathbb N$.
Then we have
\begin{equation*}
s(g)+b=(g+1)+b\overset{(2)}{=}g+(1+b)\overset{(3)}{=}g+s(b)=s(g+b)=s(b+g)=b+s(g)
\end{equation*}
So $G=\mathbb N$. Showing the initial statement is straightforward. Let arbitrary $a\in\mathbb N=G$ and the desired conclusion immediately follows.
\end{proof}

\begin{proof}(5) We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a+b\neq1)
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. Suppose for contradiction that $a+b=1$. 
We have 2 cases. First suppose $b=1$. Then $a+b=a+1=s(a)=1$, and the last equality is a contradiction of the peano postulates. Now suppose $b\neq1$. Then by lemma \ref{predecessor} there exists
$x\in\mathbb N$ such that $b=s(x)$. We then have $a+b=a+s(x)=s(a+x)=1$, which is again a contradiction.
\end{proof}
\begin{proof}(6)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a+b\neq a)
\end{equation*}
We do this by defining $G$ as the set such that
\begin{equation*}
\forall a(a\in G\leftrightarrow a\in\mathbb N\land\forall b\in\mathbb N(a+b\neq a))
\end{equation*}
(Showing existence and uniqueness is straightforward.) We show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward. We need to show $1\in G$ and $\forall g\in G(s(g)\in G)$. For the former, we already know $1\in\mathbb N$. Let arbitrary $b\in\mathbb N$.
By (5) $1+b\neq 1$.\\
\indent Now the latter. Let arbitrary $g\in G$. Then $g\in\mathbb N$ and $\forall b\in\mathbb N(g+b\neq g)$. $s(g)\in\mathbb N$ exists. Let arbitrary $b\in\mathbb N$.
For contradiction, suppose $s(g)+b=s(g)$. We then have
\begin{equation*}
s(g)=s(g)+b\overset{(4)}{=}b+s(g)=s(b+g)
\end{equation*}
By injectivity of $s$ and (4) we then have $g=b+g=g+b$. This contradicts $\forall b\in\mathbb N(g+b\neq g)$.
So $G=\mathbb N$. Showing the initial statement is straightforward. Let arbitrary $a\in\mathbb N=G$ and the desired conclusion immediately follows.
\end{proof}
\begin{proof}(7)

\forall a\in\mathbb N(a\cdot1=a=1\cdot a)

s(g)\cdot1=s(g)=s(g\cdot1)=s(1\cdot g)=(1\cdot g)+1=1\cdot s(g)
\end{equation*}
so $G=\mathbb N$. Proving the initial statement is straightforward. Let arbitrary $a\in\mathbb N=G$
and the desired conclusion immediately follows.
\end{proof}

\begin{proof}(8)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N((a+b)c=(ac)+(bc))
\end{equation*}
To do this we define $G$ as the set such that
\begin{equation*}
\forall c(c\in G\leftrightarrow c\in\mathbb N\land\forall a\in\mathbb N\forall b\in\mathbb N((a+b)c=(ac)+(bc))
\end{equation*}
(where proving existence and uniqueness is straightforward.) We show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward. We need to show $1\in G$ and $\forall g\in G(s(g)\in G)$.
For the former, we already know $1\in\mathbb N$. Let arbitrary $a,b\in\mathbb N$. We have
\begin{equation*}
(a+b)\cdot1=a+b=(a\cdot1)+(b\cdot1)
\end{equation*}
Now the latter. Let arbitrary $g\in G$. Then by definition $g\in\mathbb N$ and $\forall a\in\mathbb N\forall b\in\mathbb N((a+b)g=(ag)+(bg))$. $s(g)\in\mathbb N$ exists.

&=(as(g))+(bs(g))
\end{align*}
So $G=\mathbb N$. Proving the initial statement is straightforward. Let arbitrary $c\in\mathbb N=G$
and the desired conclusion immediately follows.

Showing $G\subseteq\mathbb N$ is straightforward. We need to show $1\in G$ and $\forall g\in G(s(g)\in G)$. For the former, we already know $1\in\mathbb N$. Let arbitrary $a,c\in\mathbb N$. We have
\begin{equation*}
(a\cdot1)c=ac\overset{(7)}{=}a(1\cdot c)
\end{equation*}
\indent Now the latter. Let arbitrary $g\in G$. Then $g\in\mathbb N$ and $\forall a\in\mathbb N\forall c\in\mathbb N((ag)c=a(gc))$. $s(g)\in\mathbb N$ exists. Let arbitrary $a,c\in\mathbb N$.
We have
\begin{align*}
(as(g))c&=((ag)+a)c\overset{(8)}{=}((ag)c)+(ac)=(a(gc))+(ac)\\
&\overset{(10)}{=}a((gc)+c)\overset{(9)}{=}a((cg)+c)=a(cs(g))\\
&\overset{(9)}{=}a(s(g)c)
\end{align*}
So $G=\mathbb N$. Proving the initial statement is straightforward. Let arbitrary $b\in\mathbb N=G$
and the desired conclusion immediately follows.
\end{proof}
\noindent(next page)\newpage
\begin{proof}(12)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N(ac=bc\to a=b)
\end{equation*}
To do this we define $G$ as the set such that
\begin{equation*}
\forall a(a\in G\leftrightarrow a\in\mathbb N\land\forall b\in\mathbb N\forall c\in\mathbb N(ac=bc\to a=b))
\end{equation*}
(where proving existence and uniqueness is straightforward.) We show that $G=\mathbb N$.
Showing $G\subseteq\mathbb N$ is straightforward. We need to show $1\in G$ and $\forall g\in G(s(g)\in G)$.\\
\indent For the former, we already know $1\in\mathbb N$. Let arbitrary $b,c\in\mathbb N$ and suppose that $1\cdot c=bc$. Suppose for contradiction that $b\neq1$. Then by lemma \ref{predecessor}
there exists $x\in\mathbb N$ such that $b=s(x)$. We then have
\begin{equation*}
c\overset{(7)}{=}1\cdot c=bc=s(x)c=(x+1)c\overset{(8,7)}{=}(xc)+c\overset{(4,9)}{=}c+(cx)
\end{equation*}
which contradicts $(6)$.\\
\indent Now the latter. Let arbitrary $g\in G$. Then $g\in\mathbb N$ and $\forall b\in\mathbb N\forall c\in\mathbb N(gc=bc\to g=b)$. $s(g)\in\mathbb N$ exists. Let arbitrary $b,c\in\mathbb N$.
Suppose $s(g)c=bc$. For contradiction, suppose $b=1$. Then
\begin{equation*}
c\overset{(7)}{=}1\cdot c=bc=s(g)c\overset{(9)}{=}cs(g)=(cg)+c\overset{(4)}{=}c+(cg)
\end{equation*}
contradicting $(6)$. So we must have $b\neq1$ and by lemma \ref{predecessor}
there exists $x\in\mathbb N$ such that $b=s(x)$. Then
\begin{equation*}
(cg)+c=cs(g)\overset{(9)}{=}s(g)c=bc=s(x)c=(x+1)c\overset{(8,7)}{=}(xc)+c
\end{equation*}
and by $(1)$, $gc=cg=xc$. Since $g\in G$ we then have $g=x$ and $s(g)=s(x)=b$.
So $G=\mathbb N$. Proving the initial statement is straightforward. Let arbitrary $a\in\mathbb N=G$
and the desired conclusion immediately follows.
\end{proof}
\begin{proof}(13)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(ab=1\leftrightarrow a=1=b)
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. First suppose $ab=1$. For contradiction suppose $b\neq1$.
Then by lemma \ref{predecessor} there exists $x\in\mathbb N$ such that $b=s(x)$. So
\begin{equation*}
1=ab=as(x)=a(x+1)\overset{(10)}{=}(ax)+(a\cdot1)=(ax)+a
\end{equation*}
which contradicts (5). So $b=1$. We immediately have $1=ab=a\cdot1=a$ and $a=1=b$. Now suppose $a=1=b$. We immediately have $ab=1\cdot1=1$.
\end{proof}

\subsection{Ordering on $\mathbb N$}
\begin{definition}
The relation $<$ on $\mathbb N$ is defined by $a<b$ if and only if there is some $p\in\mathbb N$ such that $a+p=b$ for all $a,b\in\mathbb N$.
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a<b\leftrightarrow\exists p\in\mathbb N(a+p=b))
\end{equation*}
The relation $\leq$ on $\mathbb N$ is defined by $a\leq b$ if and only if $a<b$ or $a=b$ for all $a,b\in\mathbb N$.
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a\leq b\leftrightarrow(a<b\lor a=b))
\end{equation*}
\end{definition}
\noindent We will write $a>b$ to mean the same thing as $b<a$. The same applies to $a\geq b$.
\begin{theorem}
Let $a,b\in\mathbb N$. Suppose that $a<b$. Then there is a unique $p\in\mathbb N$ such that $a+p=b$.
\end{theorem}
\begin{proof}
By definition there exists $p\in\mathbb N$ such that $a+p=b$. Let $p'\in\mathbb N$ be arbitrary such that $a+p'=b$. Then by part 4 of theorem \ref{natural arithmetic}
we have $p+a=b$ and $p'+a=b$, so $p+a=p'+a$. By part 1 of theorem \ref{natural arithmetic} we have $p=p'$.
\end{proof}
\begin{theorem}\label{natural ordering}
Let $a,b,c\in\mathbb N$.
\begin{enumerate}
\item $a\leq a$, and $a\nless a$, and $a<a+1$
\item $1\leq a$
\item If $a<b$ and $b<c$, then $a<c$; if $a\leq b$ and $b<c$, then $a<c$; if $a<b$ and $b\leq c$, then $a<c$; if $a\leq b$ and $b\leq c$, then $a\leq c$
\item $a<b$ if and only if $a+c<b+c$
\item Precisely one of $a<b$ or $a=b$ or $a>b$ holds\quad(Trichotomy Law)
\item $a<b$ if and only if $ac<bc$
\item $a\leq b$ or $b\leq a$
\item If $a\leq b$ and $b\leq a$, then $a=b$
\item It cannot be that $b<a<b+1$
\item $a\leq b$ if and only if $a<b+1$
\item $a<b$ if and only if $a+1\leq b$
\end{enumerate}
\end{theorem}
\noindent(next page)\newpage
\begin{proof}(1)
We want to show 
\begin{equation*}
\forall a\in\mathbb N(a\leq a\land a\nless a\land a<a+1)
\end{equation*}
Let arbitrary $a\in\mathbb N$. Since $a=a$, by definition $a\leq a$. Next suppose for contradiction that $a<a$. Then there would exist $p\in\mathbb N$ such that $a+p=a$, contradicting part 6
of theorem \ref{natural arithmetic}. Finally since $1\in\mathbb N$ and $a+1=a+1$, by definition $a<a+1$.
\end{proof}
\begin{proof}(2)
We want to show
\begin{equation*}
\forall a\in\mathbb N(1\leq a)
\end{equation*}
Let arbitrary $a\in\mathbb N$. We have 2 cases. First suppose $a=1$; then by definition $1\leq a$.
Next suppose $a\neq1$. Then by lemma \ref{predecessor} there exists $x\in\mathbb N$ such
that $a=s(x)=x+1$. By theorem \ref{natural arithmetic} part 4 we have $a=1+x$. So by definition $1<a$.
\end{proof}
\begin{proof}(3)
We want to show
\begin{align*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N(&[(a<b\land b<c)\to a<c]\land[(a\leq b\land b<c)\to a<c]\\
&\land[(a<b\land b\leq c)\to a<c]\land[(a\leq b\land b\leq c)\to a\leq c])
\end{align*}
Let arbitrary $a,b,c\in\mathbb N$. First suppose $a<b\land b<c$. Then there exists $x,y\in\mathbb N$ such that $a+x=b$ and $b+y=c$. We then have $(a+x)+y=b+y=c$. By
theorem \ref{natural arithmetic} part 2 we have $a+(x+y)=c$ and therefore $a<c$.\\
\indent Next suppose $a\leq b\land b<c$. There exists $y\in\mathbb N$ such that $b+y=c$.
We have 2 cases. First suppose $a<b$. Then as proven earlier we have $a<c$. Next suppose $a=b$.
Then we immediately have $a+y=c$ and $a<c$.\\
\indent Next suppose $a<b\land b\leq c$. There exists $x\in\mathbb N$ such that $a+x=b$.
We have 2 cases. First suppose $b<c$. Then again we immediately have $a<c$. Next suppose $b=c$. Then $a+x=c$ and $a<c$.\\
\indent Finally suppose $a\leq b\land b\leq c$. We have 4 cases. First suppose $a<b\land b<c$. As proven $a<c$. Next suppose $a=b\land b<c$. As proven $a<c$. 
Next suppose $a<b\land b=c$. As proven $a<c$. Finally suppose $a=b$ and $b=c$. It immediately follows that $a=c$ and $a\leq c$.
\end{proof}
\begin{proof}(4)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N(a<b\leftrightarrow a+c<b+c)
\end{equation*}
Let arbitrary $a,b,c\in\mathbb N$. First suppose $a<b$. Then there exists $x\in\mathbb N$ such that $a+x=b$. Then $(a+x)+c=b+c$ and by theorem \ref{natural arithmetic} parts 2 and 4 we have 
$(a+c)+x=b+c$. So $a+c<b+c$.\\
\indent Next suppose $a+c<b+c$. Then there exists $x\in\mathbb N$ such that $(a+c)+x=b+c$.
By theorem \ref{natural arithmetic} parts 2 and 4 we have $(a+x)+c=b+c$, and by part 1 of that same theorem we have $a+x=b$. So $a<b$.
\end{proof}
\noindent(next page)\newpage
\begin{proof}(5)
We want to prove trichotomy
\begin{align*}
\forall a\in\mathbb N&\forall b\in\mathbb N((a<b\lor a=b\lor b<a)\\
&\land\neg((a<b)\land(a=b))\land\neg((a<b)\land(b<a))\land\neg((a=b)\land(b<a)))
\end{align*}
Let arbitrary $a,b\in\mathbb N$. We start by proving the last three statements. First suppose for contradiction that $a<b$ and $a=b$. Then $a<a$, contradicting (1).
Next suppose for contradiction that $a=b$ and $b<a$. A similar argument shows that this contradicts (1).
Finally suppose for contradiction that $a<b\land b<a$. By (3) we have $a<a$, a contradiction.\\
\indent Now for the first statement. We define $G$ as the set such that
\begin{equation*}
\forall a(a\in G\leftrightarrow a\in\mathbb N\land\forall b\in\mathbb N(a<b\lor a=b\lor b<a))
\end{equation*}
(proving existence and uniqueness of $G$ is straightforward.)
We show that $G=\mathbb N$. Showing $G\subseteq\mathbb N$ is straightforward. We need to show that $1\in G$ and $\forall g\in G(s(g)\in G)$. For the former, we already know $1\in\mathbb N$. Let
arbitrary $b\in\mathbb N$. By (2) we have $1\leq b$, meaning $1<b$ or $1=b$. Either case allows us to conclude that $1\in G$.\\
\indent Now the latter. Let arbitrary $g\in G$. Then $g\in\mathbb N$ and 
$\forall b\in\mathbb N(g<b\lor g=b\lor b<g)$. $s(g)\in\mathbb N$ exists. Let arbitrary $b\in\mathbb N$. We have 3 cases. First suppose $g<b$. Then there exists $x\in\mathbb N$ 
such that $g+x=b$. Suppose $x=1$; then $s(g)=g+1=b$. Next suppose $x\neq1$; then by lemma \ref{predecessor} there exists $y\in\mathbb N$ such that $x=s(y)$. So 
$b=g+x=g+s(y)=g+(y+1)$. By theorem \ref{natural arithmetic} parts 2 and 4 we have $(g+1)+y=b$ and $s(g)<b$. For the second case suppose $g=b$. Then by (1) we have $b=g<g+1=s(g)$. 
Finally suppose $b<g$. Then since $g<g+1$ by (3) we have $b<g+1=s(g)$. We then have $G=\mathbb N$.\\
\indent Showing the desired result is straightforward. For all arbitrary $a\in\mathbb N$, $a\in G$. The result immediately follows.
\end{proof}
\begin{proof}(6)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N\forall c\in\mathbb N(ac<bc\leftrightarrow a<b)
\end{equation*}
Let arbitrary $a,b,c\in\mathbb N$. First suppose $a<b$. Then there exists $x\in\mathbb N$ such that $a+x=b$. Then by theorem \ref{natural arithmetic} part 8 we have
$bc=(a+x)c=(ac)+(xc)$; so $ac<bc$.\\
\indent Next suppose $ac<bc$. Suppose for contradiction that $a\nless b$. Then by (5) we have $a=b\lor b<a$. First suppose $a=b$. Then we immediately have $ac<ac$, which
contradicts (1). Next suppose $b<a$. There exists $x,y\in\mathbb N$ such that $(ac)+x=bc$ and $b+y=a$. By theorem \ref{natural arithmetic} parts 8 and 2 we have $bc=((b+y)c)+x=((bc)+(yc))+x=(bc)+((yc)+x)$, contradicting part 6 of that same theorem.
\end{proof}
\noindent(next page)\newpage
\begin{proof}(7) We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a\leq b\lor b\leq a)
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. First suppose $a\nless b$. Then by (5) $a=b\lor b<a$, meaning $b\leq a$. Next suppose $a<b$. Then immediately it follows that $a\leq b$.
\end{proof}
\begin{proof}(8) We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N((a\leq b\land b\leq a)\to a=b)
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. Suppose that $a\leq b\land b\leq a$. We have four cases. 
Three of the cases: $a<b\land b<a$, $a=b\land b<a$, and $a<b\land b=a$ all contradict (5). The only possibility is the last case $a=b\land b=a$.
\end{proof}
\begin{proof}(9)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(\neg(b<a\land a<b+1))
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. Suppose for contradiction that $b<a\land a<b+1$. Then there exists $x,y\in\mathbb N$ such that $b+x=a$ and $a+y=b+1$.
Then $(b+x)+y=b+1$. By theorem \ref{natural arithmetic} parts 2 and 4 we have $(x+y)+b=1+b$. By part 1 of that theorem we then have $x+y=1$, contradicting part 5 of that theorem.
\end{proof}
\begin{proof}(10)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a\leq b\leftrightarrow a<b+1)
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. First suppose $a\leq b$. Suppose for contradiction that $b+1\leq a$. Then by (3) we would have $b+1\leq b$. The case where $b+1=b$ is a contradiction to
theorem \ref{natural arithmetic} part 6. The case where $b+1<b$ implies the existence of $x\in\mathbb N$ such that $(b+1)+x=b$; by theorem \ref{natural arithmetic} part 2 we have $b+(1+x)=b$, again 
contradicting part 6 of that theorem. We can then conclude
that $\neg(b+1\leq a)$, meaning $\neg(b+1=a\lor b+1<a)$. By (5) we must then have $a<b+1$.\\
\indent Now for the other direction. Suppose $a<b+1$. Supposing that $b<a$ immediately contradicts (9). Therefore by (5) we must have $a\leq b$.
\end{proof}
\begin{proof}(11)
We want to show
\begin{equation*}
\forall a\in\mathbb N\forall b\in\mathbb N(a<b\leftrightarrow a+1\leq b)
\end{equation*}
Let arbitrary $a,b\in\mathbb N$. Suppose $a<b$. Supposing $b<a+1$ immediately contradicts (9), so we must have, by (5), $a+1\leq b$.\\
\indent Next suppose $a+1\leq b$. Suppose for contradiction that 
$b\leq a$, which would imply by (3) that $a+1\leq a$. The case where $a+1=a$ immediately contradicts theorem \ref{natural arithmetic} part 6. The case where $a+1<a$ implies the existence of 
$x\in\mathbb N$ such that $(a+1)+x=a$; theorem \ref{natural arithmetic} part 2 then implies $a+(1+x)=a$, again contradicting part 6 of that same theorem. We conclude that $\neg(b=a\lor b<a)$, so by
(5) we have $a<b$.
\end{proof}

\subsection{Well-ordering}
\begin{theorem}(Well-Ordering Principle)
Let $G\subseteq\mathbb N$ be a non-empty set. Then there is some $m\in G$ such that $m\leq g$ for all $g\in G$.
\end{theorem}
\begin{proof}
We want to show
\begin{equation*}
\forall G(G\subseteq\mathbb N\land G\neq\emptyset\to\exists m\in G(\forall g\in G(m\leq g)))
\end{equation*}
Let $G$ be arbitrary; assume that $G\subseteq\mathbb N$ and $G\neq\emptyset$. Assume for contradiction that the implication is false. That is
\begin{equation*}
\neg\exists m\in G(\forall g\in G(m\leq g))
\end{equation*}
We define $H$ as the set such that
\begin{equation*}
\forall a(a\in H\leftrightarrow a\in\mathbb N\land\forall n\in\mathbb N(n\le a\to n\notin G))
\end{equation*}
(where existence and uniqueness can be easily shown). We can show that $H\cap G=\emptyset$.
For contradiction suppose there exists $x$ such that $x\in H$ and $x\in G$. By definition then $x\in\mathbb N$ and $\forall n\in\mathbb N(n\leq x\to n\notin G)$. Then since $x\leq x$ we must have 
$x\notin G$, a contradiction. We will show that $H=\mathbb N$. Showing $H\subseteq\mathbb N$ is straightforward. We need to show $1\in H$ and $\forall h\in H(s(h)\in H)$.\\
\indent We start with the former. Suppose for contradiction that $1\notin H$. We know $1\in\mathbb N$, so we must have $\neg\forall n\in\mathbb N(n\leq 1\to n\notin G)$. This
is logically equivalent to
\begin{equation*}
\exists n\in\mathbb N(n\leq 1\land n\in G)
\end{equation*}
That is, there exists $n\in\mathbb N$ such that $n\leq 1$ and $n\in G$. $n<1$ is a contradiction as it would imply the existence of $x\in\mathbb N$ such that $n+x=1$, contradicting theorem
\ref{natural arithmetic} part 5. We therefore must have $1=n\in G$. Let arbitrary $g\in G\subseteq \mathbb N$. By theorem \ref{natural ordering} part 2 we must have $n=1\leq g$. This
contradicts our assumption that such an element does not exist, so $1\in H$.\\
\indent Now the latter. Let arbitary $h\in H$. Then $h\in\mathbb N$ and $\forall n\in\mathbb N(n\leq h\to n\notin G)$. We know $s(h)\in\mathbb N$ exists. Suppose for contradiction that 
$s(h)\notin H$. Then as before there then exists $n\in\mathbb N$ such that $n\leq s(h)$ and $n\in G$. By the contrapositive of one of the consequences
of $h\in H$, since $n\in G$ we then have $\neg(n\leq h)$,
which by theorem \ref{natural ordering} part 5 implies $h<n$. Since $n<s(h)=h+1$ would then lead to a contradiction of theorem \ref{natural ordering} part 9, we must have $n=s(h)$.
Let arbitrary $g\in G$. Suppose for contradiction that $g<n=s(h)=h+1$. Then by theorem \ref{natural ordering} part 10 we have $g\leq h$. Since $h\in H$ we then have $g\notin G$, a contradiction.
So we must have $\neg(g<n)$, which by theorem \ref{natural ordering} part 5 implies $n\leq g$.
We know that $n\in G$, so this contradicts our earlier assumption that such an element must not exist. So $s(h)\in H$ and $H=\mathbb N$.\\
\indent We know $H\cap G=\emptyset$. So $\mathbb N\cap G=\emptyset$. Suppose for contradiction that $G$ is nonempty, meaning that there exists some $x\in G$. 
Then since $G\subseteq\mathbb N$ we would have $x\in\mathbb N$ and 
therefore $x\in\mathbb N\cap G=\emptyset$, a contradiction. So we must have $G=\emptyset$, contradicting a key assumption.
\end{proof}

\section{Construction of the integers}
\subsection{Construction}
\begin{definition}
The relation $\sim$ on $\mathbb N\times\mathbb N$ is defined by $\langle a,b\rangle\sim\langle c,d\rangle$ 
if and only if $a+d=b+c$ for all $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$. That is, the set such that
\begin{equation*}
\forall x(x\in\sim\leftrightarrow\exists\langle a,b\rangle\in\mathbb N\times\mathbb N\exists\langle c,d\rangle\in\mathbb N\times\mathbb N
(x=\langle\langle a,b\rangle,\langle c,d\rangle\rangle\land a+d=b+c))
\end{equation*}
\end{definition}
\noindent(Existence follows from using $(\mathbb N\times\mathbb N)\times(\mathbb N\times\mathbb N)$ as a bounding set. 
Uniqueness from extensionality)
\begin{lemma}
The relation $\sim$ is an equivalence relation on $\mathbb N\times\mathbb N$.
\end{lemma}
\begin{proof}
Showing that $\sim\subseteq(\mathbb N\times\mathbb N)\times(\mathbb N\times\mathbb N)$ is straightforward. We need to show reflexivity on $\mathbb N\times\mathbb N$,
symmetry, and transitivity. We start with reflexivity; let arbitrary $\langle a,b\rangle\in\mathbb N\times\mathbb N$.
We immediately have $a+b=b+a$ and $\langle a,b\rangle\sim\langle a,b\rangle$. For symmetry, let $\langle a,b\rangle,\langle c,d\rangle$ be arbitrary and suppose 
$\langle a,b\rangle\sim\langle c,d\rangle$. Then by definition both pairs are members of $\mathbb N\times\mathbb N$ and $a+d=b+c$. 
We then have $c+b=b+c=a+d=d+a$, which then means $\langle c,d\rangle\sim\langle a,b\rangle$.\\
\indent Finally transitivity. Let $\langle a,b\rangle,\langle c,d\rangle,\langle e,f\rangle$ be arbitrary, suppose $\langle a,b\rangle\sim\langle c,d\rangle$ and 
$\langle c,d\rangle\sim\langle e,f\rangle$. This means that all three pairs are members of $\mathbb N\times\mathbb N$, $a+d=c+b$, and $c+f=e+d$. We then have
\begin{align*}
(a+f)+c&=a+(c+f)=a+(e+d)=(a+d)+e\\
&=(c+b)+e=(b+e)+c
\end{align*}
By cancellation we then have $a+f=b+e$. So $\langle a,b\rangle\sim\langle e,f\rangle$.
\end{proof}
\begin{definition}
The set of \textbf{integers}, denoted $\mathbb Z$, is the quotient set $(\mathbb N\times\mathbb N)/\sim$. That is,
\begin{equation*}
\forall x(x\in\mathbb Z\leftrightarrow\exists\langle a,b\rangle\in\mathbb N\times\mathbb N(x=[\langle a,b\rangle]))
\end{equation*}
where
\begin{equation*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N\forall x(x\in[\langle a,b\rangle]\leftrightarrow\langle a,b\rangle\sim x)
\end{equation*}
\end{definition}
\noindent Each integer is an equivalence class. 
Existence of each equivalence class can be shown using ran$(\sim)$ as a bounding set. The same can be done for $\mathbb Z$ using $\mathscr P(\text{ran}(\sim))$.
Uniqueness for both follows from extensionality. 
\begin{definition}
We define $\hat 0,\hat 1$ as $\hat 0=[\langle 1,1\rangle]$ and $\hat 1=[\langle 1+1,1\rangle]$.
\end{definition}

\subsection{Operations on the integers}
\begin{definition}
The binary operation $+$ on $\mathbb Z$ is defined as $+:\mathbb Z\times\mathbb Z\to\mathbb Z$ where
\begin{equation*}
\forall x(x\in+\leftrightarrow\exists\langle a,b\rangle\in\mathbb N\times\mathbb N\exists\langle c,d\rangle\in\mathbb N\times\mathbb N(x=\langle\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle,[\langle a+c,b+d\rangle]\rangle))
\end{equation*}
put simply, for all $[\langle a,b\rangle],[\langle c,d\rangle]\in\mathbb Z$,
\begin{equation*}
[\langle a,b\rangle]+[\langle c,d\rangle]=[\langle a+c,b+d\rangle]
\end{equation*}
\end{definition}
\noindent(A bounding set is $(\mathbb Z\times\mathbb Z)\times\mathbb Z$. Uniqueness follows from extensionality.)
\begin{proof}
We start by showing the domain bound dom$(+)=\mathbb Z\times\mathbb Z$. First suppose $x\in\text{dom}(+)$. Then there exists $y$ such that $\langle x,y\rangle\in+$ and
$\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such
that $x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\in\mathbb Z\times\mathbb Z$.
Next suppose $x\in\mathbb Z\times\mathbb Z$. Then there exists $x_0,x_1\in\mathbb Z$ such that
$x=\langle x_0,x_1\rangle$. There exists $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that $x_0=[\langle a,b\rangle]$
and $x_1=[\langle c,d\rangle]$. We know $\langle a+c,b+d\rangle\in\mathbb N\times\mathbb N$ and
$\langle\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle,[\langle a+c,b+d\rangle]\rangle\in+$, so
$x=\langle x_0,x_1\rangle=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\in\text{dom}(+)$.\\
\indent Now the range bound ran$(+)\subseteq\mathbb Z$. Let arbitrary $y\in\text{ran}(+)$. Then there exists $x$ such that $\langle x,y\rangle\in+$ and $a,b,c,d\in\mathbb N$ such that
$y=[\langle a+c,b+d\rangle]\in\mathbb Z$.\\
\indent Now we prove well definedness. That is, that addition between equivalence classes is independent of the particular elements used to represent each equivalence class:
\begin{align*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N&\forall\langle c,d\rangle\in\mathbb N\times\mathbb N
\forall\langle a',b'\rangle\in\mathbb N\times\mathbb N\forall\langle c',d'\rangle\in\mathbb N\times\mathbb N(\\
[\langle a,b\rangle]&=[\langle a',b'\rangle]\land[\langle c,d\rangle]=[\langle c',d'\rangle]\\
&\to[\langle a+c,b+d\rangle]=[\langle a'+c',b'+d'\rangle])
\end{align*}
Let arbitary $\langle a,b\rangle,\langle c,d\rangle,\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$ and suppose that $[\langle a,b\rangle]=[\langle a',b'\rangle]$
and $[\langle c,d\rangle]=[\langle c',d'\rangle]$. Then $\langle a,b\rangle\sim\langle a',b'\rangle$ and $a+b'=b+a'$. Similarly $c+d'=d+c'$. We then have, by associativity and commutativity of 
natural arithmetic:
\begin{align*}
(a+c)+(b'+d')&=(a+b')+(c+d')\\
&=(b+a')+(d+c')=(b+d)+(a'+c')
\end{align*}
and we can conclude that 
$\langle a+c,b+d\rangle\sim\langle a'+c',b'+d'\rangle$ and 
$[\langle a+c,b+d\rangle]=[\langle a'+c',b'+d'\rangle]$.\\
\indent Finally we show functionality. Let arbitrary $x\in\text{dom}(+)$. Then by definition there exists $y$ such that $\langle x,y\rangle\in+$. There then exists 
$\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that 
$x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle$ and $y=[\langle a+c,b+d\rangle]$. Let $y'$ be arbitrary such that 
$\langle x,y'\rangle\in+$. Then similarly there exists $\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$ such that 
$x=\langle[\langle a',b'\rangle],[\langle c',d'\rangle]\rangle$ and $y'=[\langle a'+c',b'+d'\rangle]$. 
We know $\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle=\langle[\langle a',b'\rangle],[\langle c',d'\rangle]\rangle$. So since addition is well defined we have
$y=[\langle a+c,b+d\rangle]=[\langle a'+c',b'+d'\rangle]=y'$.
\end{proof}

\begin{definition}
The binary operation $\cdot$ on $\mathbb Z$ is defined as $\cdot:\mathbb Z\times\mathbb Z\to\mathbb Z$ where
\begin{align*}
\forall x(x\in\cdot\leftrightarrow\exists\langle a,b\rangle\in&\mathbb N\times\mathbb N
\exists\langle c,d\rangle\in\mathbb N\times\mathbb N(\\&x=\langle\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle,[\langle ac+bd,ad+bc\rangle]\rangle))
\end{align*}
\end{definition}
\noindent(A bounding set is $(\mathbb Z\times\mathbb Z)\times\mathbb Z$. Uniqueness follows from extensionality.)
\begin{proof}
We start with the domain bound dom$(\cdot)=\mathbb Z\times\mathbb Z$. First suppose $x\in\text{dom}(\cdot)$. Then there exists $y$ such that $\langle x,y\rangle\in\cdot$, and 
$\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that
$x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\in\mathbb Z\times\mathbb Z$. Next suppose $x\in\mathbb Z\times\mathbb Z$. Then there exists $x_0,x_1\in\mathbb Z$ such that 
$x=\langle x_0,x_1\rangle$. There exists $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that $x_0=[\langle a,b\rangle]$ and $x_1=[\langle c,d\rangle]$.
We have $\langle\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle,[\langle ac+bd,ad+bc\rangle]\rangle\in\cdot$.
So $x=\langle x_0,x_1\rangle=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\in\text{dom}(\cdot)$.\\
\indent Next the range bound ran$(\cdot)\subseteq\mathbb Z$. Let arbitary $y\in\text{ran}(\cdot)$. 
Then there exists $x$ such that $\langle x,y\rangle\in\cdot$ and $a,b,c,d\in\mathbb N$ such that
$y=[\langle ac+bd,ad+bc\rangle]\in\mathbb Z$.\\
\indent Next we show well definedness. That is
\begin{align*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N&\forall\langle c,d\rangle\in\mathbb N\times\mathbb N\forall\langle a',b'\rangle\in\mathbb N\times\mathbb N
\forall\langle c',d'\rangle\in\mathbb N\times\mathbb N(\\
([\langle a,b\rangle]&=[\langle a',b'\rangle]\land[\langle c,d\rangle]=[\langle c',d'\rangle])\\&\to
[\langle ac+bd,ad+bc\rangle]=[\langle a'c'+b'd',a'd'+b'c'\rangle])
\end{align*}
Let arbitrary $\langle a,b\rangle,\langle c,d\rangle,\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$ and suppose  
$[\langle a,b\rangle]=[\langle a',b'\rangle]$ and $[\langle c,d\rangle]=[\langle c',d'\rangle]$.
Then we have $\langle a,b\rangle\sim\langle a',b'\rangle$ and
$a+b'=b+a'$. Similarly we have $c+d'=d+c'$. We want to show
\begin{equation*}
(ac+bd)+(a'd'+b'c')=(ad+bc)+(a'c'+b'd')
\end{equation*}
See that
\begin{align*}
&((a+b')d+(b+a')c)+((c+d')b'+(d+c')a')\\
&=((ad+b'd)+(bc+a'c))+((cb'+d'b')+(da'+c'a'))\\
&=((ad+bc)+(b'd+a'c))+((a'c'+b'd')+(da'+cb'))\\
&=((ad+bc)+(a'c'+b'd'))+((b'd+a'c)+(da'+cb'))
\end{align*}
Also see that
\begin{align*}
&((a+b')d+(b+a')c)+((c+d')b'+(d+c')a')\\
&=((b+a')d+(a+b')c)+((d+c')b'+(c+d')a')\\
&=((bd+a'd)+(ac+b'c))+((db'+c'b')+(ca'+d'a'))\\
&=((ac+bd)+(da'+cb'))+((a'd'+c'b')+(a'c+b'd))\\
&=((ac+bd)+(a'd'+c'b'))+((b'd+a'c)+(da'+cb'))
\end{align*}
By cancellation we then have our desired result, which means
\begin{equation*}
\langle ac+bd,ad+bc\rangle\sim\langle a'c'+b'd',a'd'+b'c'\rangle
\end{equation*}
and $[\langle ac+bd,ad+bc\rangle]=[\langle a'c'+b'd',a'd'+b'c'\rangle]$.\\
(next page)\newpage
\indent Finally functionality. Let arbitrary $x\in\text{dom}(\cdot)$. Then there exists $y$ such that $\langle x,y\rangle\in\cdot$ and 
$\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that 
$x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle$ and $y=[\langle ac+bd,ad+bc\rangle]$.
Let $y'$ be arbitrary such that $\langle x,y'\rangle\in\cdot$. Then similarly there exists
$\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$ such that 
$x=\langle[\langle a',b'\rangle],[\langle c',d'\rangle]\rangle$ and $y'=[\langle a'c'+b'd',a'd'+b'c'\rangle]$.
We have $\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle=\langle[\langle a',b'\rangle],[\langle c',d'\rangle]\rangle$, by well definedness we then have
$y=[\langle ac+bd,ad+bc\rangle]=[\langle a'c'+b'd',a'd'+b'c'\rangle]=y'$.
\end{proof}
\begin{definition}
The unary operation $-$ on $\mathbb Z$ is defined as $-:\mathbb Z\to\mathbb Z$ where
\begin{equation*}
\forall x(x\in-\leftrightarrow\exists\langle a,b\rangle\in\mathbb N\times\mathbb N(x=\langle[\langle a,b\rangle],[\langle b,a\rangle]))
\end{equation*}
\end{definition}
\noindent (A bounding set is $\mathbb Z\times\mathbb Z$. Uniqueness follows from extensionality.)
\begin{proof}
We start with the domain bound dom$(-)=\mathbb Z$. First let arbitrary $x\in\text{dom}(-)$. Then there exists $y$ such that $\langle x,y\rangle\in-$, and 
$\langle a,b\rangle\in\mathbb N\times\mathbb N$ such that $x=[\langle a,b\rangle]\in\mathbb Z$.
Next suppose $x\in\mathbb Z$. Then there exists $\langle a,b\rangle\in\mathbb N\times\mathbb N$ such that $x=[\langle a,b\rangle]$. 
We have $\langle[\langle a,b\rangle],[\langle b,a\rangle]\rangle\in-$. So $x=[\langle a,b\rangle]\in\text{dom}(-)$.\\
\indent Next the range bound ran$(-)\subseteq\mathbb Z$. Suppose $y\in\text{ran}(-)$. Then there exists $x$ such that $\langle x,y\rangle\in-$ and $a,b\in\mathbb N$ such that $y=[\langle b,a\rangle]\in\mathbb Z$.\\
\indent Next we show well-defineness. That is
\begin{equation*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N\forall\langle a',b'\rangle\in\mathbb N\times\mathbb N([\langle a,b\rangle]=[\langle a',b'\rangle]\to
[\langle b,a\rangle]=[\langle b',a'\rangle])
\end{equation*}
Letting arbitrary $\langle a,b\rangle,\langle a',b'\rangle\in\mathbb N\times\mathbb N$ and supposing that $[\langle a,b\rangle]=[\langle a',b'\rangle]$,
we have $a+b'=b+a'$. See that this also means $\langle b,a\rangle\sim\langle b',a'\rangle$ and therefore $[\langle b,a\rangle]=[\langle b',a'\rangle]$.\\
\indent Finally functionality. Let arbitary $x\in\text{dom}(-)$. Then there exists $y$ such that $\langle x,y\rangle\in-$, and $\langle a,b\rangle\in\mathbb N\times\mathbb N$ such that
$x=[\langle a,b\rangle]$ and $y=[\langle b,a\rangle]$.
Let $y'$ be arbitrary such that $\langle x,y'\rangle\in-$. Then there exists 
$\langle a',b'\rangle\in\mathbb N\times\mathbb N$ such that
$x=[\langle a',b'\rangle]$ and $y'=[\langle b',a'\rangle]$. We have $[\langle a,b\rangle]=[\langle a',b'\rangle]$; by well-definedness we have $y=[\langle b,a\rangle]=[\langle b',a'\rangle]=y'$.
\end{proof}

\subsection{Ordering}
\begin{definition}
The binary relation $<$ on $\mathbb Z$ is defined such that
\begin{equation*}
\forall x(x\in<\leftrightarrow\exists\langle a,b\rangle\in\mathbb N\times\mathbb N\exists\langle c,d\rangle\in\mathbb N\times\mathbb N(x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\land 
a+d<b+c))
\end{equation*}
\end{definition}
\noindent($\mathbb Z\times\mathbb Z$ is a bounding set. Uniqueness follows from extensionality.)
\begin{lemma}\label{Z_order_1}
\begin{align*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N&\forall\langle c,d\rangle\in\mathbb N\times\mathbb N\forall\langle a',b'\rangle\in\mathbb N\times\mathbb N
\forall\langle c',d'\rangle\in\mathbb N\times\mathbb N(\\(&[\langle a,b\rangle]=[\langle a',b'\rangle]\land[\langle c,d\rangle]=
[\langle c',d'\rangle])\\
&\to a+d<b+c\leftrightarrow a'+d'<b'+c')
\end{align*}
\end{lemma}
\begin{proof}
Let arbitrary $\langle a,b\rangle,\langle c,d\rangle,\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$. Suppose that 
$[\langle a,b\rangle]=[\langle a',b'\rangle]$ and $[\langle c,d\rangle]=[\langle c',d'\rangle]$, which then means $a+b'=b+a'$ and $c+d'=d+c'$. 
First suppose $a+d<b+c$. Then there exists $x$ such that $(a+d)+x=b+c$, and we have
\begin{equation*}
(((a+d)+x)+(b+a'))+(c+d')=((b+c)+(a+b'))+(d+c')
\end{equation*}
Rearranging the left side:
\begin{align*}
&(((a+d)+x)+(b+a'))+(c+d')\\
&=((a+d)+x)+((b+c)+(a'+d'))\\
&=(a+d)+(x+((b+c)+(a'+d')))\\
&=(a+d)+((b+c)+(x+(a'+d')))\\
&=((a+d)+(b+c))+((a'+d')+x) \end{align*}
and the right side:
\begin{align*}
&((b+c)+(a+b'))+(d+c')\\
&=(b+c)+((a+b')+(d+c'))\\
&=(b+c)+((a+d)+(b'+c'))\\
&=((a+d)+(b+c))+(b'+c')
\end{align*}
By cancellation we have $(a'+d')+x=b'+c'$ and $a'+d'<b'+c'$.\\
(next page)\newpage
Next suppose $a'+d'<b'+c'$. Then there exists $x$ such that $a'+d'+x=b'+c'$. We have
\begin{equation*}
(((a'+d')+x)+(a+b'))+(d+c')=((b'+c')+(b+a'))+(c+d')
\end{equation*}
Rearranging the left side:
\begin{align*}
&(((a'+d')+x)+(a+b'))+(d+c')\\
&=((a'+d')+x)+((a+b')+(d+c'))\\
&=((a'+d')+x)+((b'+c')+(a+d))\\
&=(a'+d')+(x+((b'+c')+(a+d)))\\
&=((a'+d')+(b'+c'))+((a+d)+x)
\end{align*}
and the right hand side:
\begin{align*}
&((b'+c')+(b+a'))+(c+d')\\
&=(b'+c')+((b+a')+(c+d'))\\
&=(b'+c')+((a'+d')+(b+c))\\
&=((a'+d')+(b'+c'))+(b+c)
\end{align*}
So by cancellation $(a+d)+x=b+c$ and $a+d<b+c$.
\end{proof}
\begin{lemma}\label{Z_order_2}
\begin{equation*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N\forall\langle c,d\rangle\in\mathbb N\times\mathbb N([\langle a,b\rangle]<[\langle c,d\rangle]\leftrightarrow a+d<b+c)
\end{equation*}
\end{lemma}
\begin{proof}
Let arbitrary $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$. First suppose $[\langle a,b\rangle]<[\langle c,d\rangle]$. Then by definition
there exists $\langle a_0,b_0\rangle,\langle c_0,d_0\rangle\in\mathbb N\times\mathbb N$ such that 
$[\langle a,b\rangle]=[\langle a_0,b_0\rangle]$, $[\langle c,d\rangle]=[\langle c_0,d_0\rangle]$ and $a_0+d_0<b_0+c_0$. It then immediately follows by lemma \ref{Z_order_1} that 
$a+d<b+c$.\\
\indent The reverse direction is straightforward. Suppose $a+d<b+c$ and it immediately follows by definition
that $[\langle a,b\rangle]<[\langle c,d\rangle]$.
\end{proof}
\begin{proof}(range/domain bound and well-definedness)
We start by showing $<$ is a relation on $\mathbb Z$, $<\subseteq\mathbb Z\times\mathbb Z$.
Let arbitary $x\in<$. Then by definition there exists $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that 
$x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\in\mathbb Z\times\mathbb Z$.\\
\indent Next well-definedness. That is,
\begin{align*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N&\forall\langle c,d\rangle\in\mathbb N\times\mathbb N\forall\langle a',b'\rangle\in\mathbb N\times\mathbb N
\forall\langle c',d'\rangle\in\mathbb N\times\mathbb N(\\(&[\langle a,b\rangle]=[\langle a',b'\rangle]\land[\langle c,d\rangle]=
[\langle c',d'\rangle])\\
&\to([\langle a,b\rangle]<[\langle c,d\rangle]\leftrightarrow[\langle a',b'\rangle]<[\langle c',d'\rangle]))
\end{align*}
Letting arbitary $\langle a,b\rangle,\langle c,d\rangle,\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$. Suppose that 
$[\langle a,b\rangle]=[\langle a',b'\rangle]$ and $[\langle c,d\rangle]=[\langle c',d'\rangle]$, which then means $a+b'=b+a'$ and $c+d'=d+c'$. Suppose first that 
$[\langle a,b\rangle]<[\langle c,d\rangle]$, which by lemma \ref{Z_order_2} means $a+d<b+c$. It follows by lemma \ref{Z_order_1} that $a'+d'<b'+c'$ and by lemma \ref{Z_order_2} that
$[\langle a',b'\rangle]<[\langle c',d'\rangle]$. An identical strategy proves the reverse direction.
\end{proof}
\noindent(next page)\newpage
\begin{definition}
The binary relation $\leq$ on $\mathbb Z$ is defined such that
\begin{align*}
\forall x(x\in\leq\leftrightarrow\exists\langle a,b\rangle\in\mathbb N\times\mathbb N\exists\langle c,d\rangle\in&\mathbb N\times\mathbb N(x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\\
&\land([\langle a,b\rangle]<[\langle c,d\rangle]\lor[\langle a,b\rangle]=[\langle c,d\rangle])))
\end{align*}
\end{definition}
\noindent($\mathbb Z\times\mathbb Z$ is a bounding set. Uniqueness follows from extensionality.)
\begin{lemma}\label{Z_order_3}
\begin{align*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N\forall\langle c,d\rangle\in\mathbb N\times\mathbb N(&[\langle a,b\rangle]\leq[\langle c,d\rangle]\\&\leftrightarrow
([\langle a,b\rangle]<[\langle c,d\rangle]\lor[\langle a,b\rangle]=[\langle c,d\rangle]))
\end{align*}
\end{lemma}
\begin{proof}
Let arbitrary $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$. First suppose 
$[\langle a,b\rangle]\leq[\langle c,d\rangle]$. Then by definition there exists
$\langle a_0,b_0\rangle,\langle c_0,d_0\rangle\in\mathbb N\times\mathbb N$ such that 
$[\langle a,b\rangle]=[\langle a_0,b_0\rangle]$, $[\langle c,d\rangle]=[\langle c_0,d_0\rangle]$,
and $[\langle a_0,b_0\rangle]<[\langle c_0,d_0\rangle]$ or $[\langle a_0,b_0\rangle]=[\langle c_0,d_0\rangle]$. In the first case suppose $[\langle a_0,b_0\rangle]<[\langle c_0,d_0\rangle]$. It 
then follows from the well-definedness of $<$ that $[\langle a,b\rangle]<[\langle c,d\rangle]$. In the other case suppose 
$[\langle a_0,b_0\rangle]=[\langle c_0,d_0\rangle]$. It immediately follows that $[\langle a,b\rangle]=[\langle a_0,b_0\rangle]=[\langle c_0,d_0\rangle]=[\langle c,d\rangle]$.\\
\indent Now suppose $[\langle a,b\rangle]<[\langle c,d\rangle]\lor[\langle a,b\rangle]=[\langle c,d\rangle]$. It immediately follows by definition that $[\langle a,b\rangle]\leq[\langle c,d\rangle]$.
\end{proof}
\begin{proof}(domain/range bound and well-definedness)
First we show $\leq\subseteq\mathbb Z\times\mathbb Z$. Let arbitrary $x\in\leq$.
Then there exists $\langle a,b\rangle,\langle c,d\rangle\in\mathbb N\times\mathbb N$ such that
$x=\langle[\langle a,b\rangle],[\langle c,d\rangle]\rangle\in\mathbb Z\times\mathbb Z$.\\
\indent Next well-definedness. That is
\begin{align*}
\forall\langle a,b\rangle\in\mathbb N\times\mathbb N&\forall\langle c,d\rangle\in\mathbb N\times\mathbb N\forall\langle a',b'\rangle\in\mathbb N\times\mathbb N
\forall\langle c',d'\rangle\in\mathbb N\times\mathbb N(\\(&[\langle a,b\rangle]=[\langle a',b'\rangle]\land[\langle c,d\rangle]=
[\langle c',d'\rangle])\\
&\to([\langle a,b\rangle]\leq[\langle c,d\rangle]\leftrightarrow[\langle a',b'\rangle]\leq[\langle c',d'\rangle]))
\end{align*}
Letting arbitary $\langle a,b\rangle,\langle c,d\rangle,\langle a',b'\rangle,\langle c',d'\rangle\in\mathbb N\times\mathbb N$. Suppose that 
$[\langle a,b\rangle]=[\langle a',b'\rangle]$ and $[\langle c,d\rangle]=[\langle c',d'\rangle]$. 
First suppose $[\langle a,b\rangle]\leq[\langle c,d\rangle]$. It follows by lemma \ref{Z_order_3}
that $[\langle a,b\rangle]<[\langle c,d\rangle]\lor[\langle a,b\rangle]=[\langle c,d\rangle]$.
In the first case where $[\langle a,b\rangle]<[\langle c,d\rangle]$ it follows from
the well-definedness of $<$ that $[\langle a',b'\rangle]<[\langle c',d'\rangle]$ and again by lemma \ref{Z_order_3} that $[\langle a',b'\rangle]\leq[\langle c',d'\rangle]$. For the other case where
$[\langle a,b\rangle]=[\langle c,d\rangle]$ we immediately have $[\langle a',b'\rangle]=[\langle a,b\rangle]=[\langle c,d\rangle]=[\langle c',d'\rangle]$ and by lemma \ref{Z_order_3} 
$[\langle a',b'\rangle]\leq[\langle c',d'\rangle]$. The reverse implication follows by the same argument.
\end{proof}

\subsection{Properties}
\begin{theorem}\label{integer_arithmetic_1}
Let $x,y,z\in\mathbb Z$.
\begin{enumerate}
\item$(x+y)+z=x+(y+z)$\quad(Associative Law for Addition)
\item$x+y=y+x$\quad(Commutative Law for Addition)
\item$x+\hat0=x$\quad(Identity Law for Addition)
\item$x+(-x)=\hat0$\quad(Inverses Law for Addition)
\item$(xy)z=x(yz)$\quad(Associative Law for Multiplication)
\item$xy=yx$\quad(Commutative Law for Multiplication)
\item$x\cdot\hat1=x$\quad(Identity Law for Multiplication)
\item$x(y+z)=xy+xz$\quad(Distributive Law)
\item If $xy=\hat0$, then $x=\hat0$ or $y=\hat0$\quad(No Zero Divisors Law)
\item Precisely one of $x<y$ or $x=y$ or $x>y$ holds\quad(Trichotomy Law)
\item If $x<y$ and $y<z$, then $x<z$\quad(Transitive Law)
\item If $x<y$ then $x+z<y+z$\quad(Addition Law for Order)
\item If $x<y$ and $z>\hat0$, the $xz<yz$\quad(Multiplication Law for Order)
\item $\hat0\neq\hat1$\quad(Non-Triviality)
\end{enumerate}
\end{theorem}
\begin{proof}(1)
We want to show
\begin{equation*}
\forall x\in\mathbb Z\forall y\in\mathbb Z\forall z\in\mathbb Z((x+y)+z=x+(y+z))
\end{equation*}
Let arbitrary $x,y,z\in\mathbb Z$. Then there exists $a,b,c,d,e,f\in\mathbb N$ such that 
$x=[\langle a,b\rangle]$, $y=[\langle c,d\rangle]$, and $z=[\langle e,f\rangle]$.
We have
\begin{align*}
(x+y)+z&=([\langle a,b\rangle]+[\langle c,d\rangle])+[\langle e,f\rangle]\\
&=[\langle a+c,b+d\rangle]+[\langle e,f\rangle]\\
&=[\langle (a+c)+e,(b+d)+f\rangle]\\
&=[\langle a+(c+e),b+(d+f)\rangle]\\
&=[\langle a,b\rangle]+[\langle c+e,d+f\rangle]\\
&=[\langle a,b\rangle]+([\langle c,d\rangle]+[\langle e,f\rangle])\\
&=x+(y+z)
\end{align*}
Where the fourth equality follows from associativity of $\mathbb N$.
\end{proof}
\noindent(next page)\newpage
\begin{proof}(2)
We want to show
\begin{equation*}
\forall x\in\mathbb Z\forall y\in\mathbb Z(x+y=y+x)
\end{equation*}
Let arbitrary $x,y\in\mathbb Z$. Then there exists $a,b,c,d\in\mathbb N$ such that $x=[\langle a,b\rangle]$ and $y=[\langle c,d\rangle]$. It follows that 
\begin{align*}
x+y=[\langle a,b\rangle]+[\langle c,d\rangle]&=[\langle a+c,b+d\rangle]\\
&=[\langle c+a,d+b\rangle]=[\langle c,d\rangle]+[\langle a,b\rangle]=y+x\qedhere
\end{align*}
\end{proof}
\begin{proof}(3)
We want to show
\begin{equation*}
\forall x\in\mathbb Z(x+\hat0=x)
\end{equation*}
Let arbitrary $x\in\mathbb Z$. Then there exists $a,b\in\mathbb N$ such that $x=[\langle a,b\rangle]$. We have $a+(b+1)=b+(a+1)$ and therefore $\langle a,b\rangle\sim\langle a+1,b+1\rangle$.
So
\begin{equation*}
x=[\langle a,b\rangle]=[\langle a+1,b+1\rangle]=[\langle a,b\rangle]+[\langle 1,1\rangle]=x+\hat0\qedhere
\end{equation*}
\end{proof}
\begin{proof}(4)
We want to show
\begin{equation*}
\forall x\in\mathbb Z(x+(-x)=\hat0)
\end{equation*}
Let arbitrary $x\in\mathbb Z$. Then there exists $a,b\in\mathbb N$ such that $x=[\langle a,b\rangle]$. Then by definition $-x=[\langle b,a\rangle]$.
We have $x+(-x)=[\langle a,b\rangle]+[\langle b,a\rangle]=[\langle a+b,b+a\rangle]$.
We also know $(a+b)+1=(b+a)+1$, meaning $\langle a+b,b+a\rangle\sim\langle 1,1\rangle$. Putting it all together we have
\begin{equation*}
x+(-x)=[\langle a+b,b+a\rangle]=[\langle 1,1\rangle]=\hat0\qedhere
\end{equation*}
\end{proof}
\begin{proof}(5)
We want to show
\begin{equation*}

\begin{lemma}
$\forall a\in\mathbb N\forall b\in\mathbb N([\langle a,b\rangle]=\hat0\leftrightarrow a=b)$
\end{lemma}
\begin{proof}
Let arbitary $a,b\in\mathbb N$. First suppose that $[\langle a,b\rangle]=\hat0=[\langle 1,1\rangle]$. Then $a+1=b+1$ and by cancellation $a=b$. Next suppose 
$a=b$. Then $a+1=b+1$ and $\langle a,b\rangle\sim\langle 1,1\rangle$. We then have
$[\langle a,b\rangle]=[\langle 1,1\rangle]=\hat0$.
\end{proof}
\begin{proof}(9)

\forall x\in\mathbb Z\forall y\in\mathbb Z(xy=\hat0\to(x=\hat0\lor y=\hat0))
\end{equation*}

\begin{equation*}
(ac+ad)+gd=ac+(a+g)d=ad+(a+g)c=(ac+ad)+gc
\end{equation*}
Cancellation of addition on the naturals allows us to conclude that $gd=gc$. Cancellation of multiplication allows us to conclude $c=d$. By the previous lemma this implies 
$y=[\langle c,d\rangle]=\hat0$.\\
\indent In the other case suppose $b<a$. Then there exists $g\in\mathbb N$ such that $b+g=a$.
We then have
\begin{equation*}
(bc+bd)+gc=(b+g)c+bd=(b+g)d+bc=(bc+bd)+gd
\end{equation*}
Similarly again we conclude $gc=gd$ and $c=d$, which then implies $y=[\langle c,d\rangle]=\hat0$.
\end{proof}
